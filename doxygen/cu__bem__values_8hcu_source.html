55 template <typename RangeNumberType = double>
56 void
57 shape_grad_matrix_table_assign_from_host(
58 CUDATable<4, RangeNumberType> &table_gpu,
59 const Table<2, LAPACKFullMatrixExt<RangeNumberType>> &table_cpu)
60 {
61 const unsigned int n_elements_in_grad_matrix =
62 table_cpu(0, 0).m() * table_cpu(0, 0).n();
63
64 // Get table sizes.
65 const unsigned int k3_terms = table_cpu.n_rows();
66 const unsigned int quad_num = table_cpu.n_cols();
67
68 for (unsigned int k = 0; k < k3_terms; k++)
69 {
70 for (unsigned int q = 0; q < quad_num; q++)
71 {
72 // Calculate the offset address in the table on the device.
73 std::size_t offset = (k * quad_num + q) * n_elements_in_grad_matrix;
74 table_gpu.assign_from_host(&(table_cpu(k, q).begin()->value()),
75 offset,
76 n_elements_in_grad_matrix);
77 }
78 }
79 }

97 template <typename RangeNumberType = double>
98 void
99 mapping_shape_grad_matrix_table_assign_from_host(
100 CUDATable<4, RangeNumberType> &table_gpu,
101 const Table<3, LAPACKFullMatrixExt<RangeNumberType>> &table_cpu,
102 const unsigned int max_mapping_n_shape_function)
103 {
104 // Get the sizes of table dimensions.
105 const unsigned int mapping_num = table_cpu.size(0);
106 const unsigned int k3_terms = table_cpu.size(1);
107 const unsigned int quad_num = table_cpu.size(2);
108 const unsigned int dim = table_cpu(0, 0, 0).n();
109 // The gradient matrix should have 2 columns.
110 AssertDimension(dim, 2);
111
112 // The space on GPU reserved for the gradient matrix, which can hold the
113 // data for the highest order mapping.
114 const unsigned int reserved_elements_for_grad_matrix_gpu =
115 max_mapping_n_shape_function * dim;
116
117 for (unsigned int m = 0; m < mapping_num; m++)
118 {
119 // All gradient matrices should have @p dim columns.
120 AssertDimension(dim, table_cpu(m, 0, 0).n());
121
122 // Get the gradient matrix size on CPU.
123 const unsigned int n_elements_in_grad_matrix_cpu =
124 table_cpu(m, 0, 0).m() * table_cpu(m, 0, 0).n();
125
126 Assert(reserved_elements_for_grad_matrix_gpu >=
127 n_elements_in_grad_matrix_cpu,
128 ExcInternalError());
129
130 for (unsigned int k = 0; k < k3_terms; k++)
131 {
132 for (unsigned int q = 0; q < quad_num; q++)
133 {
134 // Calculate the offset address in the table on the device.
135 std::size_t offset = ((m * k3_terms + k) * quad_num + q) *
136 reserved_elements_for_grad_matrix_gpu;
137 table_gpu.assign_from_host(
138 &(table_cpu(m, k, q).begin()->value()),
139 offset,
140 n_elements_in_grad_matrix_cpu);
141 }
142 }
143 }
144 }
