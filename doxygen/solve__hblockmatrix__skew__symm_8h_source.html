44template <typename SkewSymmetricBlockMatrixType,
45 typename VectorType,
46 typename PrecondM11Type,
47 typename PrecondM22Type,
48 typename Number>
49void
50solve_hblockmatrix_skew_symm_using_schur_complement(
51 const SkewSymmetricBlockMatrixType &block_mat,
52 VectorType &x1,
53 VectorType &x2,
54 const VectorType &b1,
55 const VectorType &b2,
56 const PrecondM11Type &precond11,
57 const PrecondM22Type &precond22,
58 const unsigned int max_iter,
59 const Number tolerance,
60 const bool log_history,
61 const bool log_result)
62{
63 // Compute \f$M_{11}^{-1} b_1\f$ and @p b4 stores the result.
64 SolverControl solver_control1(max_iter, tolerance, log_history, log_result);
65 SolverCGGeneral<VectorType> solver1(solver_control1);
66 VectorType b4(block_mat.get_M11().get_n());
67
68 solver1.solve(block_mat.get_M11(), b4, b1, precond11);
69
70 // Compute \f$M_{12}^{\mathrm{T}} M_{11}^{-1} b_1 = M_{12}^{\mathrm{T}} b_4
71 // \f$. The result vector is @p b3.
72 VectorType b3(block_mat.get_M12().get_n());
73 block_mat.get_M12().Tvmult(b3, b4);
74
75 // Compute the RHS vector in the equation for solving @p x2.
76 b3 += b2;
77
78 // Construct the Schur complement matrix and solve it for @p x2.
79 SchurComplement<SkewSymmetricBlockMatrixType,
80 VectorType,
81 PrecondM11Type,
82 Number>
83 schur_complement(
84 block_mat, precond11, max_iter, tolerance, log_history, log_result);
85 SolverControl solver_control2(max_iter, tolerance, log_history, log_result);
86 SolverCGGeneral<VectorType> solver2(solver_control2);
87 solver2.solve(schur_complement, x2, b3, precond22);
88
89 // Compute \f$-M_{12}x_2\f$ and store it into @p b4.
90 block_mat.get_M12().vmult(
91 b4, typename numbers::NumberTraits<Number>::real_type(-1.0), x2);
92 b4 += b1;
93
94 // Solve @p x1.
95 SolverControl solver_control3(max_iter, tolerance, log_history, log_result);
96 SolverCGGeneral<VectorType> solver3(solver_control3);
97 solver3.solve(block_mat.get_M11(), x1, b4, precond11);
98}
