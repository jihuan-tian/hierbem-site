81template <int dim, int spacedim, typename RangeNumberType>
82void
83assemble_fem_scaled_mass_matrix_on_one_cell(
84 const RangeNumberType factor,
85 const typename std::vector<std::pair<
86 typename DoFHandler<dim, spacedim>::active_cell_iterator,
87 typename DoFHandler<dim, spacedim>::active_cell_iterator>>::const_iterator
88 &iterator_for_cell_iterator_pairs,
89 CellWiseScratchDataForMassMatrix<dim, spacedim> &scratch_data,
90 CellWiseCopyDataForMassMatrix<dim, spacedim, RangeNumberType> &copy_data)
91{
97 copy_data.local_matrix.reinit(
98 copy_data.local_dof_indices_for_test_space.size(),
99 copy_data.local_dof_indices_for_trial_space.size());
100
110 const typename DoFHandler<dim, spacedim>::active_cell_iterator
111 &cell_iter_for_test_space_domain = iterator_for_cell_iterator_pairs->first;
112 const typename DoFHandler<dim, spacedim>::active_cell_iterator &
113 cell_iter_for_trial_space_domain = iterator_for_cell_iterator_pairs->second;
114
119 scratch_data.fe_values_for_test_space.reinit(cell_iter_for_test_space_domain);
120 scratch_data.fe_values_for_trial_space.reinit(
121 cell_iter_for_trial_space_domain);
122
123 AssertDimension(
124 scratch_data.fe_values_for_test_space.get_quadrature().size(),
125 scratch_data.fe_values_for_trial_space.get_quadrature().size());
126 const unsigned int n_q_points =
127 scratch_data.fe_values_for_test_space.get_quadrature().size();
128
129 const unsigned int dofs_per_cell_for_test_space =
130 scratch_data.fe_values_for_test_space.get_fe().dofs_per_cell;
131 const unsigned int dofs_per_cell_for_trial_space =
132 scratch_data.fe_values_for_trial_space.get_fe().dofs_per_cell;
133
141 cell_iter_for_test_space_domain->get_dof_indices(
142 copy_data.local_dof_indices_for_test_space);
143 cell_iter_for_trial_space_domain->get_dof_indices(
144 copy_data.local_dof_indices_for_trial_space);
145
150 // Iterate over test function DoFs.
151 for (unsigned int i = 0; i < dofs_per_cell_for_test_space; i++)
152 {
153 // Iterate over trial function DoFs.
154 for (unsigned int j = 0; j < dofs_per_cell_for_trial_space; j++)
155 {
156 // Iterate over each quadrature point.
157 for (unsigned int q = 0; q < n_q_points; q++)
158 {
159 Assert(
160 scratch_data.fe_values_for_test_space.JxW(q) ==
161 scratch_data.fe_values_for_trial_space.JxW(q),
162 ExcMessage(
163 "The JxW values in test space domain and trial space domain should be the same!"));
164
165 copy_data.local_matrix(i, j) +=
166 factor *
167 scratch_data.fe_values_for_test_space.shape_value(i, q) *
168 scratch_data.fe_values_for_trial_space.shape_value(j, q) *
169 scratch_data.fe_values_for_test_space.JxW(q);
170 }
171 }
172 }
173}

286template <int dim, int spacedim, typename RangeNumberType, typename MatrixType>
287void
288copy_cell_local_to_global_for_fem_matrix(
289 const CellWiseCopyDataForMassMatrix<dim, spacedim, RangeNumberType> &data,
290 MatrixType &target_matrix)
291{
292 const unsigned int dofs_per_cell_for_test_space = data.local_matrix.m();
293 const unsigned int dofs_per_cell_for_trial_space = data.local_matrix.n();
294
295 for (unsigned int i = 0; i < dofs_per_cell_for_test_space; i++)
296 {
297 for (unsigned int j = 0; j < dofs_per_cell_for_trial_space; j++)
298 {
299 target_matrix.add(data.local_dof_indices_for_test_space[i],
300 data.local_dof_indices_for_trial_space[j],
301 data.local_matrix(i, j));
302 }
303 }

370template <int dim, int spacedim>
371void
372initialize_cell_iterator_pairs_for_mass_matrix(
373 const DoFHandler<dim, spacedim> &dof_handler_for_test_space,
374 const DoFHandler<dim, spacedim> &dof_handler_for_trial_space,
375 std::vector<
376 std::pair<typename DoFHandler<dim, spacedim>::active_cell_iterator,
377 typename DoFHandler<dim, spacedim>::active_cell_iterator>>
378 &cell_iterator_pairs_for_mass_matrix)
379{
384 AssertDimension(
385 dof_handler_for_test_space.get_triangulation().n_active_cells(),
386 dof_handler_for_trial_space.get_triangulation().n_active_cells());
387
391 AssertDimension(
392 dof_handler_for_test_space.get_triangulation().n_active_cells(),
393 cell_iterator_pairs_for_mass_matrix.size());
394
395 typename DoFHandler<dim, spacedim>::active_cell_iterator
396 cell_iterator_for_test_space = dof_handler_for_test_space.begin_active();
397 typename DoFHandler<dim, spacedim>::active_cell_iterator
398 cell_iterator_for_trial_space = dof_handler_for_trial_space.begin_active();
399
400 std::size_t counter = 0;
401 for (; cell_iterator_for_test_space != dof_handler_for_test_space.end();
402 cell_iterator_for_test_space++,
403 cell_iterator_for_trial_space++,
404 counter++)
405 {
410 cell_iterator_pairs_for_mass_matrix[counter].first =
411 cell_iterator_for_test_space;
412 cell_iterator_pairs_for_mass_matrix[counter].second =
413 cell_iterator_for_trial_space;
414 }
415}

427void
428initialize_cell_iterator_pairs_for_mass_matrix(
429 const DoFHandler<dim, spacedim> &dof_handler_for_test_space,
430 const DoFHandler<dim, spacedim> &dof_handler_for_trial_space,
431 const std::set<types::material_id> &subdomain_material_ids,
432 std::vector<
433 std::pair<typename DoFHandler<dim, spacedim>::active_cell_iterator,
434 typename DoFHandler<dim, spacedim>::active_cell_iterator>>
435 &cell_iterator_pairs_for_mass_matrix)
436{
441 AssertDimension(
442 dof_handler_for_test_space.get_triangulation().n_active_cells(),
443 dof_handler_for_trial_space.get_triangulation().n_active_cells());
444
445 typename DoFHandler<dim, spacedim>::active_cell_iterator
446 cell_iterator_for_test_space = dof_handler_for_test_space.begin_active();
447 typename DoFHandler<dim, spacedim>::active_cell_iterator
448 cell_iterator_for_trial_space = dof_handler_for_trial_space.begin_active();
449
450 for (; cell_iterator_for_test_space != dof_handler_for_test_space.end();
451 cell_iterator_for_test_space++, cell_iterator_for_trial_space++)

453 auto found_iter = subdomain_material_ids.find(

455

457 cell_iterator_pairs_for_mass_matrix.push_back(
458 {cell_iterator_for_test_space, cell_iterator_for_trial_space});
459 }

484template <int dim, int spacedim, typename RangeNumberType, typename MatrixType>
485void
486assemble_fem_scaled_mass_matrix(
487 const DoFHandler<dim, spacedim> &dof_handler_for_test_space,

489 const RangeNumberType factor,
490 const Quadrature<dim> &quad_rule,
491 MatrixType &target_matrix)
492{
493 // Because the test and ansatz function spaces related to the mass matrix
494 // are on a same spatial domain, here we make an assertion about the
495 // equality of the number of cells in their respective triangulations.
496 AssertDimension(
497 dof_handler_for_test_space.get_triangulation().n_active_cells(),
498 dof_handler_for_trial_space.get_triangulation().n_active_cells());

505
506 initialize_cell_iterator_pairs_for_mass_matrix(
507 dof_handler_for_test_space,
508 dof_handler_for_trial_space,
509 cell_iterator_pairs_for_mass_matrix);
510
511 WorkStream::run(
512 cell_iterator_pairs_for_mass_matrix.begin(),
513 cell_iterator_pairs_for_mass_matrix.end(),
514 [factor](const auto &iterator_for_cell_iterator_pairs,

523 CellWiseScratchDataForMassMatrix<dim, spacedim>(
524 dof_handler_for_test_space.get_fe(),
525 dof_handler_for_trial_space.get_fe(),
526 quad_rule,
527 update_values | update_JxW_values),
528 CellWiseCopyDataForMassMatrix<dim, spacedim, RangeNumberType>(
529 dof_handler_for_test_space.get_fe(),
530 dof_handler_for_trial_space.get_fe()));
531}
