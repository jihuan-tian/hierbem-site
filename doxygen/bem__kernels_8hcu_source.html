313 const Table<3, real_type> *ky_shape_value_table = nullptr;
314 const Table<2, LAPACKFullMatrixExt<real_type>> *kx_shape_grad_matrix_table =

335 real_type Jx = 0;
336 real_type Jy = 0;
337 Tensor<1, spacedim, real_type> nx, ny;
338
342 switch (cell_neighboring_type)
343 {
344 case SamePanel: {
345 kx_shape_value_table =

347 ky_shape_value_table =
348 &(bem_values.ky_shape_value_table_for_same_panel);
349
355 if (kernel_function.kernel_type == HyperSingularRegular)

372 break;

490 Vector<real_type> kx_shape_grad_in_unit_cell(dim);
491 Vector<real_type> ky_shape_grad_in_unit_cell(dim);
492
493 for (unsigned int i = 0; i < dim; i++)
494 {
495 kx_shape_grad_in_unit_cell(i) =
496 (*kx_shape_grad_matrix_table)(k3_index, quad_no)(kx_dof_index, i);
497 ky_shape_grad_in_unit_cell(i) =
498 (*ky_shape_grad_matrix_table)(k3_index, quad_no)(ky_dof_index, i);
499 }
500
505 Vector<real_type> kx_shape_grad_in_real_cell(spacedim);
506 Vector<real_type> ky_shape_grad_in_real_cell(spacedim);
507 (*kx_covariants_table)(k3_index, quad_no)
508 .vmult(kx_shape_grad_in_real_cell, kx_shape_grad_in_unit_cell);
509 (*ky_covariants_table)(k3_index, quad_no)
510 .vmult(ky_shape_grad_in_real_cell, ky_shape_grad_in_unit_cell);
511
520 Tensor<1, spacedim, real_type> kx_shape_surface_curl =
521 dealii::cross_product_3d(
522 nx,
523 VectorToTensor<spacedim, real_type, Vector<real_type>>(
524 kx_shape_grad_in_real_cell));
525 Tensor<1, spacedim, real_type> ky_shape_surface_curl =
526 dealii::cross_product_3d(
527 ny,
528 VectorToTensor<spacedim, real_type, Vector<real_type>>(
529 ky_shape_grad_in_real_cell));
530
531 return kernel_function.value(x, y, nx, ny, component) * Jx * Jy *
532 scalar_product(kx_shape_surface_curl, ky_shape_surface_curl);
533 }
534 else
535 {
542 return kernel_function.value(x, y, nx, ny, component) * Jx * Jy *
543 (*kx_shape_value_table)(kx_dof_index, k3_index, quad_no) *
544 (*ky_shape_value_table)(ky_dof_index, k3_index, quad_no);
545 }
546 }

981 const BEMValues<dim, spacedim, real_type> &bem_values,
982 const PairCellWiseScratchData<dim, spacedim, KernelNumberType>
983 &scratch_data,
984 const unsigned int component) const
985 {
986 KernelNumberType kernel_value = KernelNumberType();
987
988 switch (cell_neighboring_type)
989 {
990 case SamePanel: {
991 // Current point in the Sauter's parametric space, at which the
992 // pulled back kernel function is to be evaluated.
993 const Point<dim * 2> &p =
994 bem_values.quad_rule_for_same_panel.point(quad_no);
995 real_type jacobian_det =
996 static_cast<real_type>(p(0) * (1 - p(0)) * (1 - p(0) * p(1)));
997
998 for (unsigned int k3_index = 0; k3_index < 8; k3_index++)
999 {
1000 kernel_value += kernel_on_unit_cell.value(
1001 k3_index, quad_no, bem_values, scratch_data, component);
1002 }
1003
1004 kernel_value *= jacobian_det;
1005
1006 break;
1007 }
1008 case CommonEdge: {
1009 // Current point in the Sauter's parametric space, at which the
1010 // pulled back kernel function is to be evaluated.
1011 const Point<dim * 2> &p =
1012 bem_values.quad_rule_for_common_edge.point(quad_no);
1013 real_type jacobian_det1 =
1014 static_cast<real_type>(p(0) * p(0) * (1 - p(0)));
1015 real_type jacobian_det2 =
1016 static_cast<real_type>(p(0) * p(0) * (1 - p(0) * p(1)));
1017
1018 kernel_value =
1019 jacobian_det1 *
1020 (kernel_on_unit_cell.value(
1021 0, quad_no, bem_values, scratch_data, component) +
1022 kernel_on_unit_cell.value(
1023 1, quad_no, bem_values, scratch_data, component)) +
1024 jacobian_det2 *
1025 (kernel_on_unit_cell.value(
1026 2, quad_no, bem_values, scratch_data, component) +
1027 kernel_on_unit_cell.value(
1028 3, quad_no, bem_values, scratch_data, component) +
1029 kernel_on_unit_cell.value(
1030 4, quad_no, bem_values, scratch_data, component) +
1031 kernel_on_unit_cell.value(
1032 5, quad_no, bem_values, scratch_data, component));
1033
1034 break;
1035 }
1036 case CommonVertex: {
1037 // Current point in the Sauter's parametric space, at which the
1038 // pulled back kernel function is to be evaluated.
1039 const Point<dim * 2> &p =
1040 bem_values.quad_rule_for_common_vertex.point(quad_no);
1041 real_type jacobian_det =
1042 static_cast<real_type>(dealii::Utilities::fixed_power<3>(p(0)));
1043
1044 for (unsigned int k3_index = 0; k3_index < 4; k3_index++)
1045 {
1046 kernel_value += kernel_on_unit_cell.value(
1047 k3_index, quad_no, bem_values, scratch_data, component);
1048 }
1049
1050 kernel_value *= jacobian_det;
1051
1052 break;
1053 }
1054 case Regular: {
1055 // There is no coordinate transformation for the regular case, so
1056 // directly evaluate the kernel function on the product of unit
1057 // cells.
1058 kernel_value = kernel_on_unit_cell.value(
1059 0, quad_no, bem_values, scratch_data, component);
1060
1061 break;
1062 }
1063 default:
1064 Assert(false, ExcInternalError());
1065 }
1066
1067 return kernel_value;
1068 }
