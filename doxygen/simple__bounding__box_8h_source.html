216template <int spacedim, typename Number>
217void
218SimpleBoundingBox<spacedim, Number>::calculate_bounding_box(
219 const std::vector<Point<spacedim, Number>> &points)
220{
224 boundary_points.first = points.at(0);
225 boundary_points.second = points.at(0);
226
230 for (const auto &point : points)
231 {
232 for (unsigned int d = 0; d < spacedim; d++)
233 {
234 if (point(d) < boundary_points.first(d))
235 {
236 boundary_points.first(d) = point(d);
237 }
238 else if (point(d) > boundary_points.second(d))
239 {
240 boundary_points.second(d) = point(d);
241 }
242 }
243 }
244}

246template <int spacedim, typename Number>
247void
248SimpleBoundingBox<spacedim, Number>::calculate_bounding_box(
249 const std::vector<types::global_dof_index> &dof_indices,
250 const std::vector<Point<spacedim, Number>> &all_support_points)
251{
256 boundary_points.first = all_support_points.at(dof_indices.at(0));
257 boundary_points.second = boundary_points.first;
258
262 for (const auto dof_index : dof_indices)
263 {
264 const Point<spacedim, Number> &point = all_support_points.at(dof_index);
265 for (unsigned int d = 0; d < spacedim; d++)
266 {
267 if (point(d) < boundary_points.first(d))
268 {
269 boundary_points.first(d) = point(d);
270 }
271 else if (point(d) > boundary_points.second(d))
272 {
273 boundary_points.second(d) = point(d);
274 }
275 }
276 }
277}

286template <int spacedim, typename Number>
287template <int dim>
288SimpleBoundingBox<spacedim, Number>::SimpleBoundingBox(
289 const Mapping<dim, spacedim> &mapping,
290 const DoFHandler<dim, spacedim> &dof_handler)
291{
292 // Allocate memory for the vector of support points, which are associated
293 // with all the DoFs in the DoFHandler.
294 std::vector<Point<spacedim, Number>> support_points(dof_handler.n_dofs());
295 // Extract the support points for all DoFs.
296 DoFTools::map_dofs_to_support_points(mapping, dof_handler, support_points);
297 // Calculate the bounding box for all the support points.
298 calculate_bounding_box(support_points);
299}

317template <int spacedim, typename Number>
318Number
319SimpleBoundingBox<spacedim, Number>::volume() const
320{
321 Number v = 1.0;
322
323 for (unsigned int d = 0; d < spacedim; d++)
324 {
325 v *= (boundary_points.second(d) - boundary_points.first(d));
326 }
327

331template <int spacedim, typename Number>
332unsigned int
333SimpleBoundingBox<spacedim, Number>::coordinate_index_with_longest_dimension()
334 const
335{
336 unsigned int coordinate_index = spacedim;
337 Number longest_dimension = 0.0;
338 for (unsigned int d = 0; d < spacedim; d++)
339 {
340 Number current_dimension =
341 boundary_points.second(d) - boundary_points.first(d);
342
343 if (current_dimension > longest_dimension)
344 {
345 longest_dimension = current_dimension;
346 coordinate_index = d;
347 }
348 }
349
350 AssertIndexRange(coordinate_index, spacedim);
351
352 return coordinate_index;
353}

355template <int spacedim, typename Number>
356std::pair<SimpleBoundingBox<spacedim, Number>,
357 SimpleBoundingBox<spacedim, Number>>
358SimpleBoundingBox<spacedim, Number>::divide_geometrically() const
359{
363 unsigned int coordinate_index = coordinate_index_with_longest_dimension();
364
365 SimpleBoundingBox<spacedim, Number> first_bbox(*this);
366 SimpleBoundingBox<spacedim, Number> second_bbox(*this);
367
372 std::pair<Point<spacedim, Number>, Point<spacedim, Number>>
373 &first_bbox_boundary_points = first_bbox.get_boundary_points();
374 first_bbox_boundary_points.second(coordinate_index) =
375 first_bbox_boundary_points.first(coordinate_index) +
376 0.5 * (first_bbox_boundary_points.second(coordinate_index) -
377 first_bbox_boundary_points.first(coordinate_index));
378
383 std::pair<Point<spacedim, Number>, Point<spacedim, Number>>
384 &second_bbox_boundary_points = second_bbox.get_boundary_points();
385 second_bbox_boundary_points.first(coordinate_index) =
386 first_bbox_boundary_points.second(coordinate_index);
387
388
389 return std::pair<SimpleBoundingBox<spacedim, Number>,
390 SimpleBoundingBox<spacedim, Number>>(first_bbox,
391 second_bbox);
392}

419template <int spacedim, typename Number>
420bool
421SimpleBoundingBox<spacedim, Number>::point_inside(
422 const Point<spacedim, Number> &p) const
423{
424 bool is_inside = true;
425
426 for (unsigned int d = 0; d < spacedim; d++)
427 {
431 if (p(d) < boundary_points.first(d) || p(d) > boundary_points.second(d))
432 {
433 is_inside = false;
434
435 break;
436 }
437 }
438
439 return is_inside;
440}
