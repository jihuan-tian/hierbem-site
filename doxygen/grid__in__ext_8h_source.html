41template <int spacedim>
42void
43assign_1d_boundary_ids(
44 const std::map<unsigned int, types::boundary_id> &boundary_ids,
45 Triangulation<1, spacedim> &triangulation)
46{
47 if (boundary_ids.size() > 0)
48 for (const auto &cell : triangulation.active_cell_iterators())
49 for (unsigned int f : GeometryInfo<1>::face_indices())
50 if (boundary_ids.find(cell->vertex_index(f)) != boundary_ids.end())
51 {
52 AssertThrow(
53 cell->at_boundary(f),
54 ExcMessage(
55 "You are trying to prescribe boundary ids on the face "
56 "of a 1d cell (i.e., on a vertex), but this face is not actually at "
57 "the boundary of the mesh. This is not allowed."));
58 cell->face(f)->set_boundary_id(
59 boundary_ids.find(cell->vertex_index(f))->second);
60 }
61}

91template <int dim, int spacedim>
92void
93read_msh(std::istream &in,
94 Triangulation<dim, spacedim> &tria,
95 const bool read_lines_as_subcelldata = false,
96 const bool reorder_cell_vertices = true,
97 const bool check_cell_orientation = false)
98{
99 AssertThrow(in.fail() == false, ExcIO());
100
101 unsigned int n_vertices;
102 unsigned int n_cells;
103 unsigned int dummy;
104 std::string line;
105 // This array stores maps from the 'entities' to the 'physical tags' for
106 // points, curves, surfaces and volumes. We use this information later to
107 // assign boundary ids.
108 std::array<std::map<int, int>, 4> tag_maps;
109
110 in >> line;
111
112 // first determine file format
113 unsigned int gmsh_file_format = 0;
114 if (line == "$NOD")
115 gmsh_file_format = 10;
116 else if (line == "$MeshFormat")
117 gmsh_file_format = 20;
118 else
119 AssertThrow(false,
120 (typename GridIn<dim, spacedim>::ExcInvalidGMSHInput(line)));
121
122 // if file format is 2.0 or greater then we also have to read the rest of
123 // the header
124 if (gmsh_file_format == 20)
125 {
126 double version;
127 unsigned int file_type, data_size;
128
129 in >> version >> file_type >> data_size;
130
131 Assert((version >= 2.0) && (version <= 4.1), ExcNotImplemented());
132 gmsh_file_format = static_cast<unsigned int>(version * 10);
133
134 Assert(file_type == 0, ExcNotImplemented());
135 Assert(data_size == sizeof(double), ExcNotImplemented());
136
137 // read the end of the header and the first line of the nodes
138 // description to synch ourselves with the format 1 handling above
139 in >> line;
140 AssertThrow(line == "$EndMeshFormat",
141 (typename GridIn<dim, spacedim>::ExcInvalidGMSHInput(line)));
142
143 in >> line;
144 // if the next block is of kind $PhysicalNames, ignore it
145 if (line == "$PhysicalNames")
146 {
147 do
148 {
149 in >> line;
150 } while (line != "$EndPhysicalNames");
151 in >> line;
152 }
153
154 // if the next block is of kind $Entities, parse it
155 if (line == "$Entities")
156 {
157 unsigned long n_points, n_curves, n_surfaces, n_volumes;
158
159 in >> n_points >> n_curves >> n_surfaces >> n_volumes;
160 for (unsigned int i = 0; i < n_points; ++i)
161 {
162 // parse point ids
163 int tag;
164 unsigned int n_physicals;
165 double box_min_x, box_min_y, box_min_z, box_max_x, box_max_y,
166 box_max_z;
167
168 // we only care for 'tag' as key for tag_maps[0]
169 if (gmsh_file_format > 40)
170 {
171 in >> tag >> box_min_x >> box_min_y >> box_min_z >>
172 n_physicals;
173 box_max_x = box_min_x;
174 box_max_y = box_min_y;
175 box_max_z = box_min_z;
176 }
177 else
178 {
179 in >> tag >> box_min_x >> box_min_y >> box_min_z >>
180 box_max_x >> box_max_y >> box_max_z >> n_physicals;
181 }
182 // if there is a physical tag, we will use it as boundary id
183 // below
184 AssertThrow(n_physicals < 2,
185 ExcMessage("More than one tag is not supported!"));
186 // if there is no physical tag, use 0 as default
187 int physical_tag = 0;
188 for (unsigned int j = 0; j < n_physicals; ++j)
189 in >> physical_tag;
190 tag_maps[0][tag] = physical_tag;
191 }
192 for (unsigned int i = 0; i < n_curves; ++i)
193 {
194 // parse curve ids
195 int tag;
196 unsigned int n_physicals;
197 double box_min_x, box_min_y, box_min_z, box_max_x, box_max_y,
198 box_max_z;
199
200 // we only care for 'tag' as key for tag_maps[1]
201 in >> tag >> box_min_x >> box_min_y >> box_min_z >> box_max_x >>
202 box_max_y >> box_max_z >> n_physicals;
203 // if there is a physical tag, we will use it as boundary id
204 // below
205 AssertThrow(n_physicals < 2,
206 ExcMessage("More than one tag is not supported!"));
207 // if there is no physical tag, use 0 as default
208 int physical_tag = 0;
209 for (unsigned int j = 0; j < n_physicals; ++j)
210 in >> physical_tag;
211 tag_maps[1][tag] = physical_tag;
212 // we don't care about the points associated to a curve, but
213 // have to parse them anyway because their format is
214 // unstructured
215 in >> n_points;
216 for (unsigned int j = 0; j < n_points; ++j)
217 in >> tag;
218 }
219
220 for (unsigned int i = 0; i < n_surfaces; ++i)
221 {
222 // parse surface ids
223 int tag;
224 unsigned int n_physicals;
225 double box_min_x, box_min_y, box_min_z, box_max_x, box_max_y,
226 box_max_z;
227
228 // we only care for 'tag' as key for tag_maps[2]
229 in >> tag >> box_min_x >> box_min_y >> box_min_z >> box_max_x >>
230 box_max_y >> box_max_z >> n_physicals;
231 // if there is a physical tag, we will use it as boundary id
232 // below
233 AssertThrow(n_physicals < 2,
234 ExcMessage("More than one tag is not supported!"));
235 // if there is no physical tag, use 0 as default
236 int physical_tag = 0;
237 for (unsigned int j = 0; j < n_physicals; ++j)
238 in >> physical_tag;
239 tag_maps[2][tag] = physical_tag;
240 // we don't care about the curves associated to a surface, but
241 // have to parse them anyway because their format is
242 // unstructured
243 in >> n_curves;
244 for (unsigned int j = 0; j < n_curves; ++j)
245 in >> tag;
246 }
247 for (unsigned int i = 0; i < n_volumes; ++i)
248 {
249 // parse volume ids
250 int tag;
251 unsigned int n_physicals;
252 double box_min_x, box_min_y, box_min_z, box_max_x, box_max_y,
253 box_max_z;
254
255 // we only care for 'tag' as key for tag_maps[3]
256 in >> tag >> box_min_x >> box_min_y >> box_min_z >> box_max_x >>
257 box_max_y >> box_max_z >> n_physicals;
258 // if there is a physical tag, we will use it as boundary id
259 // below
260 AssertThrow(n_physicals < 2,
261 ExcMessage("More than one tag is not supported!"));
262 // if there is no physical tag, use 0 as default
263 int physical_tag = 0;
264 for (unsigned int j = 0; j < n_physicals; ++j)
265 in >> physical_tag;
266 tag_maps[3][tag] = physical_tag;
267 // we don't care about the surfaces associated to a volume, but
268 // have to parse them anyway because their format is
269 // unstructured
270 in >> n_surfaces;
271 for (unsigned int j = 0; j < n_surfaces; ++j)
272 in >> tag;
273 }
274 in >> line;
275 AssertThrow(line == "$EndEntities",
276 (typename GridIn<dim, spacedim>::ExcInvalidGMSHInput(
277 line)));
278 in >> line;
279 }
280
281 // if the next block is of kind $PartitionedEntities, ignore it
282 if (line == "$PartitionedEntities")
283 {
284 do
285 {
286 in >> line;
287 } while (line != "$EndPartitionedEntities");
288 in >> line;
289 }
290
291 // but the next thing should,
292 // in any case, be the list of
293 // nodes:
294 AssertThrow(line == "$Nodes",
295 (typename GridIn<dim, spacedim>::ExcInvalidGMSHInput(line)));
296 }
297
298 // now read the nodes list
299 int n_entity_blocks = 1;
300 if (gmsh_file_format > 40)
301 {
302 int min_node_tag;
303 int max_node_tag;
304 in >> n_entity_blocks >> n_vertices >> min_node_tag >> max_node_tag;
305 }
306 else if (gmsh_file_format == 40)
307 {
308 in >> n_entity_blocks >> n_vertices;
309 }
310 else
311 in >> n_vertices;
312 std::vector<Point<spacedim>> vertices(n_vertices);
313 // set up mapping between numbering
314 // in msh-file (nod) and in the
315 // vertices vector
316 std::map<int, int> vertex_indices;
317
318 {
319 unsigned int global_vertex = 0;
320 for (int entity_block = 0; entity_block < n_entity_blocks; ++entity_block)
321 {
322 int parametric;
323 unsigned long numNodes;
324
325 if (gmsh_file_format < 40)
326 {
327 numNodes = n_vertices;
328 parametric = 0;
329 }
330 else
331 {
332 // for gmsh_file_format 4.1 the order of tag and dim is reversed,
333 // but we are ignoring both anyway.
334 int tagEntity, dimEntity;
335 in >> tagEntity >> dimEntity >> parametric >> numNodes;
336 }
337
338 std::vector<int> vertex_numbers;
339 int vertex_number;
340 if (gmsh_file_format > 40)
341 for (unsigned long vertex_per_entity = 0;
342 vertex_per_entity < numNodes;
343 ++vertex_per_entity)
344 {
345 in >> vertex_number;
346 vertex_numbers.push_back(vertex_number);
347 }
348
349 for (unsigned long vertex_per_entity = 0; vertex_per_entity < numNodes;
350 ++vertex_per_entity, ++global_vertex)
351 {
352 int vertex_number;
353 double x[3];
354
355 // read vertex
356 if (gmsh_file_format > 40)
357 {
358 vertex_number = vertex_numbers[vertex_per_entity];
359 in >> x[0] >> x[1] >> x[2];
360 }
361 else
362 in >> vertex_number >> x[0] >> x[1] >> x[2];
363
364 for (unsigned int d = 0; d < spacedim; ++d)
365 vertices[global_vertex](d) = x[d];
366 // store mapping
367 vertex_indices[vertex_number] = global_vertex;
368
369 // ignore parametric coordinates
370 if (parametric != 0)
371 {
372 double u = 0.;
373 double v = 0.;
374 in >> u >> v;
375 (void)u;
376 (void)v;
377 }
378 }
379 }
380 AssertDimension(global_vertex, n_vertices);
381 }
382
383 // Assert we reached the end of the block
384 in >> line;
385 static const std::string end_nodes_marker[] = {"$ENDNOD", "$EndNodes"};
386 AssertThrow(line == end_nodes_marker[gmsh_file_format == 10 ? 0 : 1],
387 (typename GridIn<dim, spacedim>::ExcInvalidGMSHInput(line)));
388
389 // Now read in next bit
390 in >> line;
391 static const std::string begin_elements_marker[] = {"$ELM", "$Elements"};
392 AssertThrow(line == begin_elements_marker[gmsh_file_format == 10 ? 0 : 1],
393 (typename GridIn<dim, spacedim>::ExcInvalidGMSHInput(line)));
394
395 // now read the cell list
396 if (gmsh_file_format > 40)
397 {
398 int min_node_tag;
399 int max_node_tag;
400 in >> n_entity_blocks >> n_cells >> min_node_tag >> max_node_tag;
401 }
402 else if (gmsh_file_format == 40)
403 {
404 in >> n_entity_blocks >> n_cells;
405 }
406 else
407 {
408 n_entity_blocks = 1;
409 in >> n_cells;
410 }
411
412 // set up array of cells and subcells (faces). In 1d, there is currently no
413 // standard way in deal.II to pass boundary indicators attached to
414 // individual vertices, so do this by hand via the boundary_ids_1d array
415 std::vector<CellData<dim>> cells;
416 SubCellData subcelldata;
417 std::map<unsigned int, types::boundary_id> boundary_ids_1d;
418 bool is_quad_or_hex_mesh = false;
419 bool is_tria_or_tet_mesh = false;
420
421 {
422 unsigned int global_cell = 0;
423 for (int entity_block = 0; entity_block < n_entity_blocks; ++entity_block)
424 {
425 unsigned int material_id;
426 unsigned long numElements;
427 int cell_type;
428
429 if (gmsh_file_format < 40)
430 {
431 material_id = 0;
432 cell_type = 0;
433 numElements = n_cells;
434 }
435 else if (gmsh_file_format == 40)
436 {
437 int tagEntity, dimEntity;
438 in >> tagEntity >> dimEntity >> cell_type >> numElements;
439 material_id = tag_maps[dimEntity][tagEntity];
440 }
441 else
442 {
443 // for gmsh_file_format 4.1 the order of tag and dim is reversed,
444 int tagEntity, dimEntity;
445 in >> dimEntity >> tagEntity >> cell_type >> numElements;
446 material_id = tag_maps[dimEntity][tagEntity];
447 }
448
449 for (unsigned int cell_per_entity = 0; cell_per_entity < numElements;
450 ++cell_per_entity, ++global_cell)
451 {
452 // note that since in the input
453 // file we found the number of
454 // cells at the top, there
455 // should still be input here,
456 // so check this:
457 AssertThrow(in.fail() == false, ExcIO());
458
459 unsigned int nod_num;
460
461 /*
462 For file format version 1, the format of each cell is as
463 follows: elm-number elm-type reg-phys reg-elem number-of-nodes
464 node-number-list
465
466 However, for version 2, the format reads like this:
467 elm-number elm-type number-of-tags < tag > ...
468 node-number-list
469
470 For version 4, we have:
471 tag(int) numVert(int) ...
472
473 In the following, we will ignore the element number (we simply
474 enumerate them in the order in which we read them, and we will
475 take reg-phys (version 1) or the first tag (version 2, if any
476 tag is given at all) as material id. For version 4, we already
477 read the material and the cell type in above.
478 */
479
480 unsigned int elm_number = 0;
481 if (gmsh_file_format < 40)
482 {
483 in >> elm_number // ELM-NUMBER
484 >> cell_type; // ELM-TYPE
485 }
486
487 if (gmsh_file_format < 20)
488 {
489 in >> material_id // REG-PHYS
490 >> dummy // reg_elm
491 >> nod_num;
492 }
493 else if (gmsh_file_format < 40)
494 {
495 unsigned int n_tags;
496 in >> n_tags;
497 if (n_tags >= 2)
498 {
499 // TJH: Here we read the second tag, i.e. the elementary
500 // entity tag as the material_id.
501 in >> dummy;
502 in >> material_id;
503
504 // Consume the remaining tags.
505 for (unsigned int i = 2; i < n_tags; ++i)
506 in >> dummy;
507 }
508 else if (n_tags == 1)
509 {
510 // When there is only one tag, we still use the only tag
511 // as the material_id.
512 in >> material_id;
513 }
514 else
515 material_id = 0;
516
517 if (cell_type == 1) // line
518 nod_num = 2;
519 else if (cell_type == 2) // tri
520 nod_num = 3;
521 else if (cell_type == 3) // quad
522 nod_num = 4;
523 else if (cell_type == 4) // tet
524 nod_num = 4;
525 else if (cell_type == 5) // hex
526 nod_num = 8;
527 }
528 else // file format version 4.0 and later
529 {
530 // ignore tag
531 int tag;
532 in >> tag;
533
534 if (cell_type == 1) // line
535 nod_num = 2;
536 else if (cell_type == 2) // tri
537 nod_num = 3;
538 else if (cell_type == 3) // quad
539 nod_num = 4;
540 else if (cell_type == 4) // tet
541 nod_num = 4;
542 else if (cell_type == 5) // hex
543 nod_num = 8;
544 }
545
546
547 /* `ELM-TYPE'
548 defines the geometrical type of the N-th element:
549 `1'
550 Line (2 nodes, 1 edge).
551
552 `2'
553 Triangle (3 nodes, 3 edges).
554
555 `3'
556 Quadrangle (4 nodes, 4 edges).
557
558 `4'
559 Tetrahedron (4 nodes, 6 edges, 6 faces).
560
561 `5'
562 Hexahedron (8 nodes, 12 edges, 6 faces).
563
564 `15'
565 Point (1 node).
566 */
567
568 if (((cell_type == 1) && (dim == 1)) || // a line in 1d
569 ((cell_type == 2) && (dim == 2)) || // a triangle in 2d
570 ((cell_type == 3) && (dim == 2)) || // a quadrilateral in 2d
571 ((cell_type == 4) && (dim == 3)) || // a tet in 3d
572 ((cell_type == 5) && (dim == 3))) // a hex in 3d
573 // found a cell
574 {
575 unsigned int vertices_per_cell = 0;
576 if (cell_type == 1) // line
577 vertices_per_cell = 2;
578 else if (cell_type == 2) // tri
579 {
580 vertices_per_cell = 3;
581 is_tria_or_tet_mesh = true;
582 }
583 else if (cell_type == 3) // quad
584 {
585 vertices_per_cell = 4;
586 is_quad_or_hex_mesh = true;
587 }
588 else if (cell_type == 4) // tet
589 {
590 vertices_per_cell = 4;
591 is_tria_or_tet_mesh = true;
592 }
593 else if (cell_type == 5) // hex
594 {
595 vertices_per_cell = 8;
596 is_quad_or_hex_mesh = true;
597 }
598
599 AssertThrow(nod_num == vertices_per_cell,
600 ExcMessage(
601 "Number of nodes does not coincide with the "
602 "number required for this object"));
603
604 // allocate and read indices
605 cells.emplace_back();
606 cells.back().vertices.resize(vertices_per_cell);
607 for (unsigned int i = 0; i < vertices_per_cell; ++i)
608 {
609 // hypercube cells need to be reordered
610 if (vertices_per_cell ==
611 GeometryInfo<dim>::vertices_per_cell)
612 {
613 in >> cells.back()
614 .vertices[GeometryInfo<dim>::ucd_to_deal[i]];
615 }
616 else
617 {
618 in >> cells.back().vertices[i];
619 }
620 }
621
622 // to make sure that the cast won't fail
623 Assert(material_id <=
624 std::numeric_limits<types::material_id>::max(),
625 ExcIndexRange(
626 material_id,
627 0,
628 std::numeric_limits<types::material_id>::max()));
629 // we use only material_ids in the range from 0 to
630 // numbers::invalid_material_id-1
631 AssertIndexRange(material_id, numbers::invalid_material_id);
632
633 cells.back().material_id = material_id;
634
635 // transform from gmsh to consecutive numbering
636 for (unsigned int i = 0; i < vertices_per_cell; ++i)
637 {
638 AssertThrow(
639 vertex_indices.find(cells.back().vertices[i]) !=
640 vertex_indices.end(),
641 (typename GridIn<dim, spacedim>::
642 ExcInvalidVertexIndexGmsh(cell_per_entity,
643 elm_number,
644 cells.back().vertices[i])));
645
646 // vertex with this index exists
647 cells.back().vertices[i] =
648 vertex_indices[cells.back().vertices[i]];
649 }
650 }
651 else if ((cell_type == 1) &&
652 ((dim == 2) || (dim == 3))) // a line in 2d or 3d
653 // boundary info
654 {
655 if (read_lines_as_subcelldata)
656 {
657 subcelldata.boundary_lines.emplace_back();
658 in >> subcelldata.boundary_lines.back().vertices[0] >>
659 subcelldata.boundary_lines.back().vertices[1];
660
661
662 // to make sure that the cast won't fail
663 Assert(material_id <=
664 std::numeric_limits<types::boundary_id>::max(),
665 ExcIndexRange(
666 material_id,
667 0,
668 std::numeric_limits<types::boundary_id>::max()));
669 // we use only boundary_ids in the range from 0 to
670 // numbers::internal_face_boundary_id-1
671 AssertIndexRange(material_id,
672 numbers::internal_face_boundary_id);
673
674 subcelldata.boundary_lines.back().boundary_id =
675 static_cast<types::boundary_id>(material_id);
676
677 // transform from ucd to
678 // consecutive numbering
679 for (unsigned int &vertex :
680 subcelldata.boundary_lines.back().vertices)
681 if (vertex_indices.find(vertex) != vertex_indices.end())
682 // vertex with this index exists
683 vertex = vertex_indices[vertex];
684 else
685 {
686 // no such vertex index
687 AssertThrow(false,
688 (typename GridIn<dim, spacedim>::
689 ExcInvalidVertexIndex(cell_per_entity,
690 vertex)));
691 vertex = numbers::invalid_unsigned_int;
692 }
693 }
694 else
695 {
696 // TJH: Here we only read the data file but do not add any
697 // lines to the @p subcelldata, since in the current BEM
698 // solver, the boundary of the boundary of a volume is
699 // empty.
700 unsigned int v1, v2;
701 in >> v1 >> v2;
702 }
703 }
704 else if ((cell_type == 2 || cell_type == 3) &&
705 (dim == 3)) // a triangle or a quad in 3d
706 // boundary info
707 {
708 unsigned int vertices_per_cell = 0;
709 // check cell type
710 if (cell_type == 2) // tri
711 {
712 vertices_per_cell = 3;
713 is_tria_or_tet_mesh = true;
714 }
715 else if (cell_type == 3) // quad
716 {
717 vertices_per_cell = 4;
718 is_quad_or_hex_mesh = true;
719 }
720
721 subcelldata.boundary_quads.emplace_back();
722
723 // resize vertices
724 subcelldata.boundary_quads.back().vertices.resize(
725 vertices_per_cell);
726 // for loop
727 for (unsigned int i = 0; i < vertices_per_cell; ++i)
728 in >> subcelldata.boundary_quads.back().vertices[i];
729
730 // to make sure that the cast won't fail
731 Assert(material_id <=
732 std::numeric_limits<types::boundary_id>::max(),
733 ExcIndexRange(
734 material_id,
735 0,
736 std::numeric_limits<types::boundary_id>::max()));
737 // we use only boundary_ids in the range from 0 to
738 // numbers::internal_face_boundary_id-1
739 AssertIndexRange(material_id,
740 numbers::internal_face_boundary_id);
741
742 subcelldata.boundary_quads.back().boundary_id =
743 static_cast<types::boundary_id>(material_id);
744
745 // transform from gmsh to
746 // consecutive numbering
747 for (unsigned int &vertex :
748 subcelldata.boundary_quads.back().vertices)
749 if (vertex_indices.find(vertex) != vertex_indices.end())
750 // vertex with this index exists
751 vertex = vertex_indices[vertex];
752 else
753 {
754 // no such vertex index
755 Assert(
756 false,
757 (typename GridIn<dim, spacedim>::ExcInvalidVertexIndex(
758 cell_per_entity, vertex)));
759 vertex = numbers::invalid_unsigned_int;
760 }
761 }
762 else if (cell_type == 15)
763 {
764 // read the indices of nodes given
765 unsigned int node_index = 0;
766 if (gmsh_file_format < 20)
767 {
768 // For points (cell_type==15), we can only ever
769 // list one node index.
770 AssertThrow(nod_num == 1, ExcInternalError());
771 in >> node_index;
772 }
773 else
774 {
775 in >> node_index;
776 }
777
778 // we only care about boundary indicators assigned to
779 // individual vertices in 1d (because otherwise the vertices
780 // are not faces)
781 if (dim == 1)
782 boundary_ids_1d[vertex_indices[node_index]] = material_id;
783 }
784 else
785 {
786 AssertThrow(
787 false,
788 (typename GridIn<dim, spacedim>::ExcGmshUnsupportedGeometry(
789 cell_type)));
790 }
791 }
792 }
793 AssertDimension(global_cell, n_cells);
794 }
795 // Assert that we reached the end of the block
796 in >> line;
797 static const std::string end_elements_marker[] = {"$ENDELM", "$EndElements"};
798 AssertThrow(line == end_elements_marker[gmsh_file_format == 10 ? 0 : 1],
799 (typename GridIn<dim, spacedim>::ExcInvalidGMSHInput(line)));
800
801 // check that no forbidden arrays are used
802 Assert(subcelldata.check_consistency(dim), ExcInternalError());
803
804 AssertThrow(in.fail() == false, ExcIO());
805
806 // check that we actually read some cells.
807 AssertThrow(cells.size() > 0,
808 (typename GridIn<dim, spacedim>::ExcGmshNoCellInformation(
809 subcelldata.boundary_lines.size(),
810 subcelldata.boundary_quads.size())));
811
812 // TODO: the functions below (GridTools::delete_unused_vertices(),
813 // GridTools::invert_all_negative_measure_cells(),
814 // GridTools::consistently_order_cells()) need to be revisited
815 // for simplex/mixed meshes
816
817 if (dim == 1 || (is_quad_or_hex_mesh && !is_tria_or_tet_mesh))
818 {
819 // do some clean-up on vertices...
820 GridTools::delete_unused_vertices(vertices, cells, subcelldata);
821 // ... and cells
822 if (dim == spacedim)
823 GridTools::invert_cells_with_negative_measure(vertices, cells);
824 if (reorder_cell_vertices)
825 GridTools::consistently_order_cells(cells);
826 }
827 else if (is_tria_or_tet_mesh)
828 {
829 if (dim == spacedim)
830 GridTools::invert_cells_with_negative_measure(vertices, cells);
831 }
832
833 if (check_cell_orientation)
834 tria.create_triangulation(vertices, cells, subcelldata);
835 else
836 tria.create_triangulation_without_orientation_checking(vertices,
837 cells,
838 subcelldata);
839
840 // in 1d, we also have to attach boundary ids to vertices, which does not
841 // currently work through the call above
842 if (dim == 1)
843 assign_1d_boundary_ids(boundary_ids_1d, tria);
844}

863template <int dim, int spacedim>
864void
865read_msh(const std::string &mesh_file,
866 Triangulation<dim, spacedim> &tria,
867 const bool check_cell_orientation = true)
868{
869 // gmsh -> deal.II types
870 const std::map<int, std::uint8_t> gmsh_to_dealii_type = {
871 {{15, 0}, {1, 1}, {2, 2}, {3, 3}, {4, 4}, {7, 5}, {6, 6}, {5, 7}}};
872
873 // Vertex renumbering, by dealii type
874 const std::array<std::vector<unsigned int>, 8> gmsh_to_dealii = {
875 {{0},
876 {{0, 1}},
877 {{0, 1, 2}},
878 {{0, 1, 3, 2}},
879 {{0, 1, 2, 3}},
880 {{0, 1, 3, 2, 4}},
881 {{0, 1, 2, 3, 4, 5}},
882 {{0, 1, 3, 2, 4, 5, 7, 6}}}};
883
884 std::vector<Point<spacedim>> vertices;
885 std::vector<CellData<dim>> cells;
886 SubCellData subcelldata;
887 std::map<unsigned int, types::boundary_id> boundary_ids_1d;
888
889 gmsh::initialize();
890 gmsh::option::setNumber("General.Verbosity", 0);
891 gmsh::open(mesh_file);
892
893 AssertThrow(gmsh::model::getDimension() == dim,
894 ExcMessage("You are trying to read a gmsh file with dimension " +
895 std::to_string(gmsh::model::getDimension()) +
896 " into a grid of dimension " + std::to_string(dim)));
897
898 // Read all nodes, and store them in our vector of vertices. Before we do
899 // that, make sure all tags are consecutive
900 {
901 gmsh::model::mesh::removeDuplicateNodes();
902 gmsh::model::mesh::renumberNodes();
903 std::vector<std::size_t> node_tags;
904 std::vector<double> coord;
905 std::vector<double> parametricCoord;
906 gmsh::model::mesh::getNodes(
907 node_tags, coord, parametricCoord, -1, -1, false, false);
908 vertices.resize(node_tags.size());
909 for (unsigned int i = 0; i < node_tags.size(); ++i)
910 {
911 // Check that renumbering worked!
912 AssertDimension(node_tags[i], i + 1);
913 for (unsigned int d = 0; d < spacedim; ++d)
914 vertices[i][d] = coord[i * 3 + d];
915#ifdef DEBUG
916 // Make sure the embedded dimension is right
917 for (unsigned int d = spacedim; d < 3; ++d)
918 Assert(std::abs(coord[i * 3 + d]) < 1e-10,
919 ExcMessage("The grid you are reading contains nodes that are "
920 "nonzero in the coordinate with index " +
921 std::to_string(d) +
922 ", but you are trying to save "
923 "it on a grid embedded in a " +
924 std::to_string(spacedim) + " dimensional space."));
925#endif
926 }
927 }
928
929 // TJH: Get all the elementary entities of dimension @p dim in the model, as a
930 // vector of (dimension, tag) pairs. This is different from
931 // dealii::read_msh(const std::string &fname), which read entities of all
932 // dimensions.
933 std::vector<std::pair<int, int>> entities;
934 gmsh::model::getEntities(entities, dim);
935
936 for (const auto &e : entities)
937 {
938 // Dimension and tag of the entity:
939 const int &entity_dim = e.first;
940 const int &entity_tag = e.second;
941
942 types::manifold_id manifold_id = numbers::flat_manifold_id;
943 types::boundary_id boundary_id = 0;
944
945 // Get the physical tags, to deduce boundary, material, and
946 // manifold_id
947 std::vector<int> physical_tags;
948 gmsh::model::getPhysicalGroupsForEntity(entity_dim,
949 entity_tag,
950 physical_tags);
951
952 // Now fill manifold id and boundary or material id
953 if (physical_tags.size())
954 for (auto physical_tag : physical_tags)
955 {
956 std::string name;
957 gmsh::model::getPhysicalName(entity_dim, physical_tag, name);
958 if (!name.empty())
959 try
960 {
961 std::map<std::string, int> id_names;
962 Patterns::Tools::to_value(name, id_names);
963 bool throw_anyway = false;
964 bool found_boundary_id = false;
965 // If the above did not throw, we keep going, and retrieve
966 // all the information that we know how to translate.
967 for (const auto &it : id_names)
968 {
969 const auto &name = it.first;
970 const auto &id = it.second;
971 if (entity_dim == dim && name == "MaterialID")
972 {
973 boundary_id = static_cast<types::boundary_id>(id);
974 found_boundary_id = true;
975 }
976 else if (entity_dim < dim && name == "BoundaryID")
977 {
978 boundary_id = static_cast<types::boundary_id>(id);
979 found_boundary_id = true;
980 }
981 else if (name == "ManifoldID")
982 manifold_id = static_cast<types::manifold_id>(id);
983 else
984 // We did not recognize one of the keys. We'll fall
985 // back to setting the boundary id to the physical
986 // tag after reading all strings.
987 throw_anyway = true;
988 }
989 // If we didn't find a BoundaryID:XX or MaterialID:XX, and
990 // something was found but not recognized, then we set the
991 // material id or boundary id in the catch block below,
992 // using directly the physical tag
993 if (throw_anyway && !found_boundary_id)
994 throw;
995 }
996 catch (...)
997 {
998 // When the above didn't work, we revert to the old
999 // behaviour: the physical tag itself is interpreted
1000 // either as a material_id or a boundary_id, and no
1001 // manifold id is known
1002 boundary_id = physical_tag;
1003 }
1004 }
1005
1006 // Get the mesh elements for the entity (dim, tag):
1007 std::vector<int> element_types;
1008 std::vector<std::vector<std::size_t>> element_ids, element_nodes;
1009 gmsh::model::mesh::getElements(
1010 element_types, element_ids, element_nodes, entity_dim, entity_tag);
1011
1012 for (unsigned int i = 0; i < element_types.size(); ++i)
1013 {
1014 const auto &type = gmsh_to_dealii_type.at(element_types[i]);
1015 const auto n_vertices = gmsh_to_dealii[type].size();
1016 const auto &elements = element_ids[i];
1017 const auto &nodes = element_nodes[i];
1018 for (unsigned int j = 0; j < elements.size(); ++j)
1019 {
1020 if (entity_dim == dim)
1021 {
1022 cells.emplace_back(n_vertices);
1023 auto &cell = cells.back();
1024 for (unsigned int v = 0; v < n_vertices; ++v)
1025 cell.vertices[v] =
1026 nodes[n_vertices * j + gmsh_to_dealii[type][v]] - 1;
1027 cell.manifold_id = manifold_id;
1028 // TJH: Assign the @p entity_tag as @p material_id, which will be
1029 // used to collect cells belong to a surface in DDM.
1030 cell.material_id = entity_tag;
1031 }
1032 else if (entity_dim == 2)
1033 {
1034 subcelldata.boundary_quads.emplace_back(n_vertices);
1035 auto &face = subcelldata.boundary_quads.back();
1036 for (unsigned int v = 0; v < n_vertices; ++v)
1037 face.vertices[v] =
1038 nodes[n_vertices * j + gmsh_to_dealii[type][v]] - 1;
1039
1040 face.manifold_id = manifold_id;
1041 face.boundary_id = boundary_id;
1042 }
1043 else if (entity_dim == 1)
1044 {
1045 subcelldata.boundary_lines.emplace_back(n_vertices);
1046 auto &line = subcelldata.boundary_lines.back();
1047 for (unsigned int v = 0; v < n_vertices; ++v)
1048 line.vertices[v] =
1049 nodes[n_vertices * j + gmsh_to_dealii[type][v]] - 1;
1050
1051 line.manifold_id = manifold_id;
1052 line.boundary_id = boundary_id;
1053 }
1054 else if (entity_dim == 0)
1055 {
1056 // TJH: This is only effective in one dimension. In the
1057 // original function dealii::read_msh(const std::string
1058 // &fname), here is an assertion about @p dim==1. Since a
1059 // mesh generated from a CAD model (when there are no
1060 // physical groups defined, "save all" should be enabled in
1061 // Gmsh to save the mesh, otherwise, nothing will be saved)
1062 // contains geometric entities of all dimensions, such as a
1063 // vertex with zero dimension, it will make the original
1064 // assertion fail.
1065 if (dim == 1)
1066 for (unsigned int j = 0; j < elements.size(); ++j)
1067 boundary_ids_1d[nodes[j] - 1] = boundary_id;
1068 }
1069 }
1070 }
1071 }
1072
1073 Assert(subcelldata.check_consistency(dim), ExcInternalError());
1074
1075 if (check_cell_orientation)
1076 tria.create_triangulation(vertices, cells, subcelldata);
1077 else
1078 tria.create_triangulation_without_orientation_checking(vertices,
1079 cells,
1080 subcelldata);
1081
1082 // in 1d, we also have to attach boundary ids to vertices, which does not
1083 // currently work through the call above
1084 if (dim == 1)
1085 assign_1d_boundary_ids(boundary_ids_1d, tria);
1086
1087 gmsh::clear();
1088 gmsh::finalize();
1089}
