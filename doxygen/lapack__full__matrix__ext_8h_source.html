1544template <typename Number>
1545void
1546balance_frobenius_norm(LAPACKFullMatrixExt<Number> &A,
1547 LAPACKFullMatrixExt<Number> &B)
1548{
1549 using real_type = typename numbers::NumberTraits<Number>::real_type;
1550
1551 real_type norm_A = A.frobenius_norm();
1552 real_type norm_B = B.frobenius_norm();
1553 real_type factor = std::sqrt(norm_A / norm_B);
1554
1555 if (std::abs(factor) >= std::numeric_limits<real_type>::epsilon())
1556 {
1557 A /= factor;
1558 }
1559
1560 B *= factor;
1561}

1660template <typename Number>
1661typename LAPACKFullMatrixExt<Number>::size_type
1662LAPACKFullMatrixExt<Number>::reduced_svd_on_AxBH(
1663 LAPACKFullMatrixExt<Number> &A,
1664 LAPACKFullMatrixExt<Number> &B,
1665 LAPACKFullMatrixExt<Number> &U,
1666 std::vector<real_type> &Sigma_r,
1667 LAPACKFullMatrixExt<Number> &VT,
1668 real_type singular_value_threshold)
1669{
1681 AssertDimension(A.n(), B.n());
1682
1683 const size_type mm = A.m();
1684 const size_type nn = B.m();
1685 const size_type min_dim = std::min(mm, nn);
1686
1687 bool is_qr_used;
1688
1694 const size_type formal_rank = A.n();
1695
1696 if (A.m() > formal_rank && B.m() > formal_rank)
1697 {
1698 is_qr_used = true;
1699
1705 LAPACKFullMatrixExt<Number> QA, RA;
1706 A.reduced_qr(QA, RA);
1707
1712 LAPACKFullMatrixExt<Number> QB, RB;
1713 B.reduced_qr(QB, RB);
1714
1719 LAPACKFullMatrixExt<Number> R, U_hat, VT_hat;
1724 R.reinit(RA.m(), RB.m());
1725 RA.mHmult(R, RB);
1726 R.svd(U_hat, Sigma_r, VT_hat);
1727
1728 U.reinit(QA.m(), formal_rank);
