55template <int dim, int spacedim>
56void
57write_msh_correct(const Triangulation<dim, spacedim> &tria, std::ostream &out)
58{
59 AssertThrow(out, ExcIO());
60
61 // get the positions of the
62 // vertices and whether they are
63 // used.
64 const std::vector<Point<spacedim>> &vertices = tria.get_vertices();
65 const std::vector<bool> &vertex_used = tria.get_used_vertices();
66
67 const unsigned int n_vertices = tria.n_used_vertices();
68
69 typename Triangulation<dim, spacedim>::active_cell_iterator cell =
70 tria.begin_active();
71 const typename Triangulation<dim, spacedim>::active_cell_iterator endc =
72 tria.end();
73
74 // Write Header
75 // The file format is:
76 /*
77 $NOD
78 number-of-nodes
79 node-number x-coord y-coord z-coord
80 ...
81 $ENDNOD
82 $ELM
83 number-of-elements
84 elm-number elm-type number-of-tags reg-phys reg-elem node-number-list
85 ...
86 $ENDELM
87 */
88 out << "$MeshFormat\n"
89 << "2.2 0 8\n"
90 << "$EndMeshFormat\n"
91 << "$Nodes" << '\n'
92 << n_vertices << '\n';
93
94 // actually write the vertices.
95 // note that we shall number them
96 // with first index 1 instead of 0
97 for (unsigned int i = 0; i < vertices.size(); ++i)
98 if (vertex_used[i])
99 {
100 out << i + 1 // vertex index
101 << " " << vertices[i];
102 for (unsigned int d = spacedim + 1; d <= 3; ++d)
103 out << " 0"; // fill with zeroes
104 out << '\n';
105 }
106
107 // Write cells preamble
108 out << "$EndNodes" << '\n'
109 << "$Elements" << '\n'
110 << tria.n_active_cells() << '\n';
111
112 /*
113 elm-type
114 defines the geometrical type of the n-th element:
115 1
116 Line (2 nodes).
117 2
118 Triangle (3 nodes).
119 3
120 Quadrangle (4 nodes).
121 4
122 Tetrahedron (4 nodes).
123 5
124 Hexahedron (8 nodes).
125 6
126 Prism (6 nodes).
127 7
128 Pyramid (5 nodes).
129 8
130 Second order line (3 nodes: 2 associated with the vertices and 1 with the
131 edge).
132 9
133 Second order triangle (6 nodes: 3 associated with the vertices and 3 with
134 the edges). 10 Second order quadrangle (9 nodes: 4 associated with the
135 vertices, 4 with the edges and 1 with the face). 11 Second order
136 tetrahedron (10 nodes: 4 associated with the vertices and 6 with the
137 edges). 12 Second order hexahedron (27 nodes: 8 associated with the
138 vertices, 12 with the edges, 6 with the faces and 1 with the volume). 13
139 Second order prism (18 nodes: 6 associated with the vertices, 9 with the
140 edges and 3 with the quadrangular faces). 14 Second order pyramid (14
141 nodes: 5 associated with the vertices, 8 with the edges and 1 with the
142 quadrangular face). 15 Point (1 node).
143 */
144 unsigned int elm_type;
145 switch (dim)
146 {
147 case 1:
148 elm_type = 1;
149 break;
150 case 2:
151 elm_type = 3;
152 break;
153 case 3:
154 elm_type = 5;
155 break;
156 default:
157 Assert(false, ExcNotImplemented());
158 break;
159 }
160
161 // write cells. Enumerate cells
162 // consecutively, starting with 1
163 const unsigned int number_of_tags = 3;
164 for (cell = tria.begin_active(); cell != endc; ++cell)
165 {
166 out << cell->active_cell_index() + 1 << ' ' << elm_type << ' '
167 << number_of_tags << ' '
168 << static_cast<unsigned int>(cell->material_id()) << ' '
169 << static_cast<unsigned int>(cell->material_id()) << ' '
170 << static_cast<unsigned int>(cell->subdomain_id()) << ' ';
171
172 // Vertex numbering follows UCD conventions.
173
174 for (unsigned int vertex = 0;
175 vertex < GeometryInfo<dim>::vertices_per_cell;
176 ++vertex)
177 out << cell->vertex_index(GeometryInfo<dim>::ucd_to_deal[vertex]) + 1
178 << ' ';
179 out << '\n';
180 }
181
182 out << "$EndElements\n";
183
184 // make sure everything now gets to
185 // disk
186 out.flush();
187
188 AssertThrow(out, ExcIO());
189}
