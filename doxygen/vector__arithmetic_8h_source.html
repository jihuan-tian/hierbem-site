41template <typename Value>
42void
43add_vector(std::vector<Value> &x,
44 const std::vector<Value> &y,
45 const Value alpha)
46{
47 AssertDimension(x.size(), y.size());
48
49 for (size_t i = 0; i < x.size(); i++)
50 {
51 x[i] += alpha * y[i];
52 }
53}

63template <typename Value>
64void
65scale_vector(std::vector<Value> &x, const Value alpha)
66{
67 for (auto &v : x)

69 v *= alpha;

73template <typename Value>
74void
75normalize_vector(std::vector<Value> &x)
76{
77 Value magnitude = Value();
78 for (auto v : x)
79 magnitude += v * v;
80 magnitude = std::sqrt(magnitude);
81
82 scale_vector(x, magnitude);
83}
