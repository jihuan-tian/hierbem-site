76template <typename VectorType>
77template <typename MatrixType, typename PreconditionerType>
78void
79SolverCGGeneral<VectorType>::solve(const MatrixType &A,
80 VectorType &x,
81 const VectorType &b,
82 const PreconditionerType &preconditioner)
83{
84 // At the moment, @p LAPACKFullMatrixExt or @p dealii::SparseMatrix uses
85 // @p m() and @p n() to get matrix sizes while @p HMaxtrix uses @p get_m()
86 // and @p get_n(). Because they are not consistent, we do not perform a
87 // dimension assertion about the matrix here.
88 AssertDimension(x.size(), b.size());
89 LogStream::Prefix prefix("cg-general");
90
91 const size_type n = b.size();
92
93 // Compute the residual vector \f$Ax^0-b\f$.
94 VectorType r(b);
95 // TODO Vector scaling should be generalized to handle more vector types.
96 BLASHelpers::scal_helper(n, Number(-1.0), r);
97 A.vmult_add(r, Number(1.0), x);
98
99 // Preconditioned residual vector.
100 VectorType v(n);
101 preconditioner.vmult(v, r);
102
103 // Correction vector
104 VectorType p(v);
105 VectorType p_prev(p);
106
107 Number rho0 = LinAlg::inner_product_tbb(v, r);
108 Number rho_prev = rho0;
109 Number rho = rho0;
110
111 if (control.log_result())
112 {
113 deallog << "step=0,residual_norm="
114 << numbers::NumberTraits<Number>::abs(rho0) << std::endl;
115 }
116
117 unsigned int i = 1;
118 for (; i <= control.max_steps(); i++)
119 {
120 // Transformed correction vector.
121 VectorType s(n);
122 A.vmult(s, p);
123 Number sigma = LinAlg::inner_product_tbb(s, p);
124 // Step size in the correction direction.
125 Number alpha = rho / sigma;
126 // Update the solution vector.
127 x.add(-alpha, p);
128 // Update the residual vector.
129 r.add(-alpha, s);
130 // Update the preconditioned residual vector.
131 preconditioner.vmult(v, r);
132 // Update rho.
133 rho = LinAlg::inner_product_tbb(v, r);
134
135 if (control.log_history())
136 {
137 deallog << "step=" << i << ",residual_norm="
138 << numbers::NumberTraits<Number>::abs(rho) << std::endl;
139 }
140
141 if (numbers::NumberTraits<Number>::abs(rho) <=
142 control.tolerance() * numbers::NumberTraits<Number>::abs(rho0))
143 {
144 // CG is convergent.
145 return;
146 }
147 else
148 {
149 // Update the correction vector.
150 Number beta = rho / rho_prev;
151 p *= beta;
152 p += v;
153 rho_prev = rho;
154 }
155 }
156
157 AssertThrow(
158 false,
159 SolverControl::NoConvergence(i, numbers::NumberTraits<Number>::abs(rho)));
160}
