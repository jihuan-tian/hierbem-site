66 template <int dim, int spacedim>
67 void
68 extract_subdomain_dofs(const DoFHandler<dim, spacedim> &dof_handler,
69 const std::set<types::subdomain_id> &subdomain_ids,
70 std::vector<bool> &selected_dofs)
71 {
72 Assert(selected_dofs.size() == dof_handler.n_dofs(),
73 ExcDimensionMismatch(selected_dofs.size(), dof_handler.n_dofs()));
74
75 // preset all values by false
76 std::fill_n(selected_dofs.begin(), dof_handler.n_dofs(), false);
77
78 // Global DoF indices for the current cell.
79 std::vector<types::global_dof_index> cell_dof_indices;
80 cell_dof_indices.reserve(
81 dof_handler.get_fe_collection().max_dofs_per_cell());
82
83 // this function is similar to the make_sparsity_pattern function, see
84 // there for more information
85 typename DoFHandler<dim, spacedim>::active_cell_iterator
86 cell = dof_handler.begin_active(),
87 endc = dof_handler.end();
88 for (; cell != endc; ++cell)
89 {
90 // Find the current cell's subdomain id in the given list.
91 auto found_iter = subdomain_ids.find(cell->subdomain_id());
92
93 if (found_iter != subdomain_ids.end())
94 {
95 const unsigned int dofs_per_cell = cell->get_fe().dofs_per_cell;
96 cell_dof_indices.resize(dofs_per_cell);
97 cell->get_dof_indices(cell_dof_indices);
98 for (unsigned int i = 0; i < dofs_per_cell; ++i)
99 selected_dofs[cell_dof_indices[i]] = true;
100 }
101 }
102 }

433 template <int dim, int spacedim>
434 types::global_dof_index
435 extract_material_domain_dofs_excluding_boundary_dofs(
436 const DoFHandler<dim, spacedim> &dof_handler,
437 const std::set<types::material_id> &boundary_cell_material_ids,
438 std::vector<bool> &selected_dofs)
439 {
440 AssertDimension(selected_dofs.size(), dof_handler.n_dofs());
441 types::global_dof_index effective_n_dofs = dof_handler.n_dofs();
442
443 // preset all values by true.
444 std::fill_n(selected_dofs.begin(), dof_handler.n_dofs(), true);
445
446 std::vector<types::global_dof_index> face_dof_indices(
447 dof_handler.get_fe().dofs_per_face);
448
449 typename DoFHandler<dim, spacedim>::active_cell_iterator
450 cell = dof_handler.begin_active(),
451 endc = dof_handler.end();
452 for (; cell != endc; ++cell)
453 {
454 // Find the current cell's material id in the list of ids for the
455 // boundary cells.
456 auto found_iter = boundary_cell_material_ids.find(cell->material_id());
457
458 if (found_iter != boundary_cell_material_ids.end())
459 {
460 // Iterate over each face, which is actually a line in
461 // this case, and check if it completely lies at boundary.
462 for (unsigned int f = 0; f < GeometryInfo<dim>::faces_per_cell; f++)
463 {
464 // Get the face iterator as a @p DoFAccessor, but it
465 // should be converted to a @p CellAccessor to call
466 // @p at_boundary.
467 if (cell->at_boundary(f))
468 {
469 // Get the face iterator as a @p DoFCellAccessor.
470 cell->face(f)->get_dof_indices(face_dof_indices);
471
472 for (unsigned int d = 0;
473 d < dof_handler.get_fe().dofs_per_face;
474 d++)
475 {
476 if (selected_dofs[face_dof_indices[d]])
477 {
478 selected_dofs[face_dof_indices[d]] = false;
479 effective_n_dofs--;
480 }
481 }
482 }
483 }
484 }
485 }
486
487 return effective_n_dofs;
488 }

583 template <int dim, int spacedim, typename Number = double>
584 void
585 map_dofs_to_support_points_and_normal_vectors(
586 MappingInfo<dim, spacedim> &mapping_info,
587 const DoFHandler<dim, spacedim> &dof_handler,
588 std::vector<Point<spacedim, Number>> &support_points,
589 std::vector<Tensor<1, spacedim, Number>> &normal_vectors,
590 const bool is_normal_vector_negated = false)
591 {
592 const types::global_dof_index n_dofs = dof_handler.n_dofs();
593 const unsigned n_dofs_per_cell = dof_handler.get_fe().dofs_per_cell;
594 AssertDimension(n_dofs, support_points.size());
595 AssertDimension(n_dofs, normal_vectors.size());
596
597 // Flags indicating if a support point or normal vector at a DoF support
598 // point has been computed.
599 std::vector<bool> dof_flags;
600 dof_flags.assign(n_dofs, false);
601
602 // Get the unit support points of the finite element space.
603 const std::vector<Point<dim>> &fe_unit_support_points =
604 dof_handler.get_fe().get_unit_support_points();
605
606 std::unique_ptr<typename MappingQ<dim, spacedim>::InternalData>
607 &mapping_data = mapping_info.get_data();
608 const unsigned int mapping_n_shape_functions =
609 mapping_data->n_shape_functions;
610 mapping_info.resize_internal_data(n_dofs_per_cell);
611
612 // Compute mapping shape function values and their derivatives at the
613 // support points of the finite element.
614 mapping_data->compute_shape_function_values(fe_unit_support_points);
615
616 // Collect gradient matrices of mapping shape functions, which are evaluated
617 // at the finite element support points.
618 std::vector<LAPACKFullMatrixExt<Number>> mapping_grad_matrices(
619 n_dofs_per_cell);
620 for (unsigned int d = 0; d < n_dofs_per_cell; d++)
621 {
622 mapping_grad_matrices[d].reinit(mapping_n_shape_functions, dim);
623 BEMTools::mappingq_shape_grad_matrix<dim, spacedim, Number>(
624 *mapping_data, d, mapping_grad_matrices[d]);
625 }
626
627 // Iterate over each cell in the DoF handler.
628 std::vector<types::global_dof_index> cell_dof_indices(n_dofs_per_cell);
629 for (const auto &cell : dof_handler.active_cell_iterators())
630 {
631 // Compute the support points of the mapping in the current cell.
632 mapping_info.get_mapping().compute_mapping_support_points(cell);
633 const std::vector<Point<spacedim>> &mapping_support_points =
634 mapping_info.get_mapping().get_support_points();
635 // Get the DoF indices in the current cell.
636 cell->get_dof_indices(cell_dof_indices);
637
638 for (unsigned int d = 0; d < n_dofs_per_cell; d++)
639 {
640 const types::global_dof_index dof_index = cell_dof_indices[d];
641 if (!dof_flags[dof_index])
642 {
643 // Compute support point.
644 const Point<spacedim> fe_support_point =
645 mapping_info.get_mapping().transform_unit_to_real_cell(
646 fe_unit_support_points[d]);
647 support_points[dof_index] = fe_support_point;
648
649 // Compute normal vector.
650 Tensor<1, spacedim> normal_vector;
651 BEMTools::surface_jacobian_det_and_normal_vector(
652 mapping_support_points,
653 mapping_grad_matrices[d],
654 normal_vector,
655 is_normal_vector_negated);
656 normal_vectors[dof_index] = normal_vector;
657
658 dof_flags[dof_index] = true;
659 }
660 }
661 }
662 }
