109template <typename SkewSymmetricBlockMatrixType,
110 typename VectorType,
111 typename PrecondM11Type,
112 typename Number>
113void
114SchurComplement<SkewSymmetricBlockMatrixType,
115 VectorType,
116 PrecondM11Type,
117 Number>::vmult(VectorType &y, const VectorType &x) const
118{
119 // It is possible that M12 is a non-symmetric H-matrix, whose vmult function
120 // is accumulative. Therefore, we set the result vector as zero first to clear
121 // previous values.
122 block_matrix.get_M12().vmult(*y1, x);
123 SolverControl solver_control(max_iter, tolerance, log_history, log_result);
124 SolverCGGeneral<VectorType> solver(solver_control);
125 solver.solve(block_matrix.get_M11(), *y2, *y1, precond11);
126 block_matrix.get_M12().Tvmult(*y3, *y2);
127 block_matrix.get_M22().vmult(y, x);
128 y += *y3;
129}

144{

146 vmult(y_tmp, x);
147 BLASHelpers::scal_helper(y_tmp.size(),
148 typename VectorType::value_type(alpha),
149 y_tmp);
150 y += y_tmp;
151}
