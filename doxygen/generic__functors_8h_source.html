85template <typename VectorType>
86void
87permute_vector_by_ipiv(VectorType &v,
88 const std::vector<dealii::types::blas_int> &ipiv)
89{
90 AssertDimension(v.size(), ipiv.size());
91
92 typename VectorType::value_type temp;
93 for (std::size_t i = 0; i < ipiv.size(); i++)
94 {
95 temp = v[i];
96 v[i] = v[ipiv[i] - 1];
97 v[ipiv[i] - 1] = temp;
98 }
99}

281template <int spacedim, typename Number = double>
282bool
283is_equal(const Point<spacedim, Number> &p1,
284 const Point<spacedim, Number> &p2,
285 const Number threshold)
286{
287 Assert(threshold > 0, ExcMessage("The given threshold value should be >0!"));
288
289 bool equality = true;
290
291 for (unsigned int i = 0; i < spacedim; i++)
292 {
293 if (std::abs(p1(i) - p2(i)) > threshold)
294 {
295 equality = false;
296 break;
297 }
298 }
299
300 return equality;
301}

406template <typename Value>
407void
408intersect(const std::array<Value, 2> &range1,
409 const std::array<Value, 2> &range2,
410 std::array<Value, 2> &range_intersection)
411{
412 range_intersection[0] = 0;
413 range_intersection[1] = 0;
414
415 types::global_dof_index larger_lower_bound = std::max(range1[0], range2[0]);
416 types::global_dof_index smaller_upper_bound = std::min(range1[1], range2[1]);
417
418 if (smaller_upper_bound > larger_lower_bound)
419 {
420 // When a non-empty range is obtained.
421 range_intersection[0] = larger_lower_bound;
422 range_intersection[1] = smaller_upper_bound;
423 }
424}
