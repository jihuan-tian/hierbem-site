147template <int dim, int spacedim>
148Point<spacedim>
149MappingQExt<dim, spacedim>::transform_unit_to_real_cell(
150 const Point<dim> &p) const
151{
152 // Set up the polynomial space in the lexicographic order.
153 const TensorProductPolynomials<dim> tensor_pols(
154 Polynomials::generate_complete_Lagrange_basis(this->line_support_points));
155 Assert(tensor_pols.n() ==
156 Utilities::fixed_power<dim>(this->polynomial_degree + 1),
157 ExcInternalError());
158
159 // Get the numbering for accessing the support points in the lexicographic
160 // ordering which are stored in the hierarchic ordering.
161 const std::vector<unsigned int> renumber(
162 FETools::lexicographic_to_hierarchic_numbering<dim>(
163 this->polynomial_degree));
164
165 Point<spacedim> mapped_point;
166 for (unsigned int i = 0; i < tensor_pols.n(); ++i)
167 mapped_point +=
168 support_points[renumber[i]] * tensor_pols.compute_value(i, p);
169
170 return mapped_point;
171}
