36template <template <typename> typename VectorType, typename real_type>
37class ComplexOutputDataVector
38{
39public:
40 template <int dim,
41 int spacedim,
42 template <typename>
43 typename VectorType1,
44 typename real_type1>
45 friend void
46 add_complex_data_vector(
47 DataOut<dim, spacedim> &data_out,
48 const DoFHandler<dim, spacedim> &dof_handler,
49 const ComplexOutputDataVector<VectorType1, real_type1> &complex_data,
50 const std::string variable_name);
51
52 ComplexOutputDataVector(const VectorType<std::complex<real_type>> &vec)
53 : real_part(vec.size())
54 , imag_part(vec.size())
55 , amplitude(vec.size())
56 , angle(vec.size())
57 {
58 LinAlg::get_vector_real_part(real_part, vec);
59 LinAlg::get_vector_imag_part(imag_part, vec);
60 LinAlg::get_vector_amplitude(amplitude, vec);
61 LinAlg::get_vector_angle(angle, vec);
62 }
63
64private:
65 VectorType<real_type> real_part;
66 VectorType<real_type> imag_part;
67 VectorType<real_type> amplitude;
68 VectorType<real_type> angle;
69};

72template <int dim,
73 int spacedim,
74 template <typename>
75 typename VectorType,
76 typename real_type>
77void
78add_complex_data_vector(
79 DataOut<dim, spacedim> &data_out,
80 const DoFHandler<dim, spacedim> &dof_handler,
81 const ComplexOutputDataVector<VectorType, real_type> &complex_data,
82 const std::string variable_name)
83{
84 data_out.add_data_vector(dof_handler,
85 complex_data.real_part,
86 variable_name + std::string("_real"));
87 data_out.add_data_vector(dof_handler,
88 complex_data.imag_part,
89 variable_name + std::string("_imag"));
90 data_out.add_data_vector(dof_handler,
91 complex_data.amplitude,
92 variable_name + std::string("_amplitude"));
93 data_out.add_data_vector(dof_handler,
94 complex_data.angle,
95 variable_name + std::string("_angle"));
96}
