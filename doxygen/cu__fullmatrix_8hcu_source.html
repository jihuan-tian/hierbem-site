627 template <typename T>
628 HBEM_ATTR_HOST HBEM_ATTR_DEV void
629 CUDAFullMatrix<T>::invert_by_gauss_elim(CUDAFullMatrix<T> &M_inv)
630 {
631 assert(rows == cols);
632 assert(rows == M_inv.m());
633 assert(cols == M_inv.n());
634
635 M_inv.reinit();
636
640 for (size_type l = 0; l < rows; l++)
641 {
646 M_inv(l, l) = 1.0 / (*this)(l, l);
647
657 for (size_type j = l + 1; j < rows; j++)
658 {
659 (*this)(l, j) = M_inv(l, l) * (*this)(l, j);
660 }
661
670 for (size_type j = 0; j < l; j++)
671 {
672 M_inv(l, j) = M_inv(l, l) * M_inv(l, j);
673 }
674
679 for (size_type i = l + 1; i < rows; i++)
680 {
685 for (size_type j = 0; j <= l; j++)
686 {
687 M_inv(i, j) = M_inv(i, j) - (*this)(i, l) * M_inv(l, j);
688 }
689
694 for (size_type j = l + 1; j < rows; j++)
695 {
696 (*this)(i, j) = (*this)(i, j) - (*this)(i, l) * (*this)(l, j);
697 }
698 }
699 }
700
705 for (size_type l = rows - 1; l > 0; l--)
706 {
710 size_type i = l - 1;
728 while (true)
729 {
735 for (size_type j = 0; j < rows; j++)
736 {
737 M_inv(i, j) = M_inv(i, j) - (*this)(i, l) * M_inv(l, j);
738 }
739
740 if (i == 0)
741 {
742 break;
743 }
744 else
745 {
746 i--;
747 }
748 }
749 }
750 }

753 template <typename T>
754 HBEM_ATTR_HOST HBEM_ATTR_DEV void
755 CUDAFullMatrix<T>::invert2x2(const CUDAFullMatrix<T> &M)
756 {
757 assert(rows == cols);
758 assert(rows == M.m());
759 assert(cols == M.n());
760
761 (*this)(0, 0) = M(1, 1);
762 (*this)(0, 1) = -M(0, 1);
763 (*this)(1, 0) = -M(1, 0);
764 (*this)(1, 1) = M(0, 0);
765
766 (*this) /= M.determinant2x2();
767 }

804 template <typename T>
805 HBEM_ATTR_HOST HBEM_ATTR_DEV T
806 CUDAFullMatrix<T>::determinant3x3() const
807 {
808 return (*this)(0, 0) * (*this)(1, 1) * (*this)(2, 2) +
809 (*this)(0, 1) * (*this)(1, 2) * (*this)(2, 0) +
810 (*this)(0, 2) * (*this)(1, 0) * (*this)(2, 1) -
811 (*this)(0, 2) * (*this)(1, 1) * (*this)(2, 0) -
812 (*this)(0, 1) * (*this)(1, 0) * (*this)(2, 2) -
813 (*this)(0, 0) * (*this)(1, 2) * (*this)(2, 1);
