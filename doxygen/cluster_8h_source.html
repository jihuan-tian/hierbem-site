693template <int spacedim, typename Number = double>
694Number
695calc_cluster_distance(
696 const Cluster<spacedim, Number> &cluster1,
697 const Cluster<spacedim, Number> &cluster2,
698 const std::vector<Point<spacedim, Number>> &all_support_points)
699{
700 Number cluster_distance =
701 all_support_points.at(cluster1.get_index_set().at(0))
702 .distance(all_support_points.at(cluster2.get_index_set().at(0)));
703 Number point_pair_distance;
704
705 for (const auto &index1 : cluster1.get_index_set())
706 {
707 for (const auto &index2 : cluster2.get_index_set())
708 {
709 point_pair_distance = all_support_points.at(index1).distance(
710 all_support_points.at(index2));
711
712 if (point_pair_distance < cluster_distance)
713 {
714 cluster_distance = point_pair_distance;
715 }
716 }
717 }
718
719 return cluster_distance;
720}

920template <int spacedim, typename Number = double>
921Number
922calc_cluster_distance(
923 const Cluster<spacedim, Number> &cluster1,
924 const Cluster<spacedim, Number> &cluster2,
925 const std::vector<Point<spacedim, Number>> &all_support_points,
926 const std::vector<Number> &cell_size_at_dofs)
927{
931 Number uncorrected_cluster_distance =
932 calc_cluster_distance(cluster1, cluster2, all_support_points);
933
938 Number max_dof_cell_size = 0.0;
939
940 for (const auto &index : cluster1.get_index_set())
941 {
942 if (cell_size_at_dofs.at(index) > max_dof_cell_size)
943 {
944 max_dof_cell_size = cell_size_at_dofs.at(index);
945 }
946 }
947
948 for (const auto &index : cluster2.get_index_set())
949 {
950 if (cell_size_at_dofs.at(index) > max_dof_cell_size)
951 {
952 max_dof_cell_size = cell_size_at_dofs.at(index);
953 }
954 }
955
956 Number distance_correction = max_dof_cell_size * 2;
957
961 if (uncorrected_cluster_distance > distance_correction)
962 {
963 return uncorrected_cluster_distance - distance_correction;
964 }
965 else
966 {
967 return uncorrected_cluster_distance;
968 }
969}
