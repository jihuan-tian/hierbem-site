81Quadrature<dim>::Quadrature(const Point<dim> &point)
82 : quadrature_points(std::vector<Point<dim>>(1, point))
83 , weights(std::vector<double>(1, 1.))
84 , is_tensor_product_flag(true)
85 , tensor_basis(new std::array<Quadrature<1>, dim>())
86{

88 {

95template <int dim>
96Quadrature<dim>::Quadrature(const SubQuadrature &q1, const Quadrature<1> &q2)
97 : quadrature_points(q1.size() * q2.size())
98 , weights(q1.size() * q2.size())
99 , is_tensor_product_flag(q1.is_tensor_product())
100{
101 unsigned int present_index = 0;
102 for (unsigned int i2 = 0; i2 < q2.size(); ++i2)
103 for (unsigned int i1 = 0; i1 < q1.size(); ++i1)
104 {
105 // compose coordinates of new quadrature point by tensor product in the
106 // last component
107 for (unsigned int d = 0; d < dim - 1; ++d)
108 quadrature_points[present_index](d) = q1.point(i1)(d);
109 quadrature_points[present_index](dim - 1) = q2.point(i2)(0);
110
111 weights[present_index] = q1.weight(i1) * q2.weight(i2);
112
113 ++present_index;
114 }
115
116#ifdef DEBUG
117 if (size() > 0)
118 {
119 double sum = 0;
120 for (unsigned int i = 0; i < size(); ++i)
121 sum += weights[i];
122 // we cannot guarantee the sum of weights to be exactly one, but it should
123 // be near that.
124 Assert((sum > 0.999999) && (sum < 1.000001), ExcInternalError());
125 }
126#endif
127
128 if (is_tensor_product_flag)
129 {
130 tensor_basis = std::make_unique<std::array<Quadrature<1>, dim>>();
131 for (unsigned int i = 0; i < dim - 1; ++i)
132 (*tensor_basis)[i] = q1.get_tensor_basis()[i];
133 (*tensor_basis)[dim - 1] = q2;
134 }
135}

138template <int dim>
139Quadrature<dim>::Quadrature(const Quadrature<dim != 1 ? 1 : 0> &q)
140 : Subscriptor()
141 , quadrature_points(Utilities::fixed_power<dim>(q.size()))
142 , weights(Utilities::fixed_power<dim>(q.size()))
143 , is_tensor_product_flag(true)
144{
145 Assert(dim <= 4, ExcNotImplemented());
146
147 const unsigned int n0 = q.size();
148 const unsigned int n1 = (dim > 1) ? n0 : 1;
149 const unsigned int n2 = (dim > 2) ? n0 : 1;
150 const unsigned int n3 = (dim > 3) ? n0 : 1;
151
152 unsigned int k = 0;
153
154 // N.B. The following embedded loops show that the first coordinate component
155 // runs the fastest.
156 for (unsigned int i3 = 0; i3 < n3; ++i3)
157 for (unsigned int i2 = 0; i2 < n2; ++i2)
158 for (unsigned int i1 = 0; i1 < n1; ++i1)
159 for (unsigned int i0 = 0; i0 < n0; ++i0)
160 {
161 quadrature_points[k](0) = q.point(i0)(0);
162 if (dim > 1)
163 quadrature_points[k](1) = q.point(i1)(0);
164 if (dim > 2)
165 quadrature_points[k](2) = q.point(i2)(0);
166 if (dim > 3)
167 quadrature_points[k](3) = q.point(i3)(0);
168
169 weights[k] = q.weight(i0);
170 if (dim > 1)
171 weights[k] *= q.weight(i1);
172 if (dim > 2)
173 weights[k] *= q.weight(i2);
174 if (dim > 3)
175 weights[k] *= q.weight(i3);
176
177 ++k;
178 }
179
180 tensor_basis = std::make_unique<std::array<Quadrature<1>, dim>>();
181 for (unsigned int i = 0; i < dim; ++i)
182 (*tensor_basis)[i] = q;
183}

199template <int dim>
200Quadrature<dim> &
201Quadrature<dim>::operator=(const Quadrature<dim> &q)
202{
203 weights = q.weights;
204 quadrature_points = q.quadrature_points;
205 is_tensor_product_flag = q.is_tensor_product_flag;
206 if (dim > 1 && is_tensor_product_flag)
207 {
208 if (tensor_basis == nullptr)
209 tensor_basis =
210 std::make_unique<std::array<Quadrature<1>, dim>>(*q.tensor_basis);
211 else
212 *tensor_basis = *q.tensor_basis;
213 }
214 return *this;
215}
