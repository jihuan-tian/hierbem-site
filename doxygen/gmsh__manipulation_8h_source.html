154template <int dim, int spacedim>
155bool
156GmshManip<dim, spacedim>::is_point_in_volume(const EntityTag volume_tag,
157 const std::vector<double> &coord)
158{
159 AssertDimension(coord.size(), spacedim);
160
161 if (dim == 1)
162 {
163 for (unsigned int i = dim + 1; i < spacedim; i++)
164 Assert(coord[i] == 0, ExcInternalError());
165 }
166
167 if (gmsh::model::isInside(dim + 1, volume_tag, coord, false) > 0)
168 {
169 return true;
170 }
171 else
172 {
173 return false;
174 }
175}

178template <int dim, int spacedim>
179void
180GmshManip<dim, spacedim>::get_normal(
181 const EntityTag entity_tag,
182 const std::vector<double> &parametric_coord,
183 std::vector<double> &normal)
184{
185 AssertDimension(normal.size(), spacedim);
186 AssertDimension(parametric_coord.size(), dim);
187
188 switch (dim)
189 {
190 case 1: {
191 // Compute the tangent vector on the curve.
192 std::vector<double> tangent;
193 gmsh::model::getDerivative(dim,
194 entity_tag,
195 parametric_coord,
196 tangent);
197
198 for (unsigned int i = dim + 1; i < spacedim; i++)
199 Assert(tangent[i] == 0, ExcInternalError());
200
201 // Normalize the tangent vector.
202 normalize_vector(tangent);
203
204 // Rotate the tangent vector by \f$-\frac{\pi}{2}\f$ to get the
205 // normal vector.
206 normal[0] = tangent[1];
207 normal[1] = -tangent[0];
208 for (unsigned int i = dim + 1; i < spacedim; i++)
209 normal[i] = 0;
210
211 break;
212 }
213 case 2: {
214 gmsh::model::getNormal(entity_tag, parametric_coord, normal);
215 break;
216 }
217 default: {
218 Assert(false, ExcInternalError());
219 break;
220 }
221 }
222}

225template <int dim, int spacedim>
226void
227GmshManip<dim, spacedim>::get_surface_normal_at_cell_barycenter(
228 const EntityTag surface_tag,
229 std::vector<double> &normal,
230 std::vector<double> &root_point)
231{
232 AssertDimension(normal.size(), spacedim);
233 AssertDimension(root_point.size(), spacedim);
234
235 std::vector<int> element_types;
236 std::vector<std::vector<std::size_t>> element_tags;
237 std::vector<std::vector<std::size_t>> node_tags;
238
239 gmsh::model::mesh::getElements(
240 element_types, element_tags, node_tags, dim, surface_tag);
241
242 // Ensure that we only have one type of elements.
243 AssertDimension(element_types.size(), 1);
244
245 // Get barycenters of all cells on the surface.
246 std::vector<double> barycenters;
247 gmsh::model::mesh::getBarycenters(
248 element_types[0], surface_tag, false, false, barycenters);
249
250 // We only use the first barycenter to get the normal vector.
251 std::vector<double> first_barycenter(spacedim);
252 for (unsigned int i = 0; i < spacedim; i++)
253 {
254 first_barycenter[i] = barycenters[i];
255 }
256
257 if (dim == 1)
258 {
259 for (unsigned int i = dim + 1; i < spacedim; i++)
260 Assert(first_barycenter[i] == 0, ExcInternalError());
261 }
262
263 std::vector<double> parametric_coord(dim);
264 gmsh::model::getParametrization(dim,
265 surface_tag,
266 first_barycenter,
267 parametric_coord);
268 get_normal(surface_tag, parametric_coord, normal);
269
270 // Because the cell is flat while the surface may be curved, the barycenter
271 // may not lie exactly on the surface. Hence, we transform the t (for
272 // @p dim=1) or uv (for @p dim=2) local coordinates to Cartesian frame,
273 // which inherently perform projection onto the surface.
274 gmsh::model::getValue(dim, surface_tag, parametric_coord, root_point);
275}

278template <int dim, int spacedim>
279int
280GmshManip<dim, spacedim>::get_surface_intrinsic_orientation(
281 const EntityTag surface_tag,
282 const EntityTag volume_tag,
283 const double eps)
284{
285 std::vector<double> normal(spacedim);
286 std::vector<double> root_point(spacedim);
287
288 get_surface_normal_at_cell_barycenter(surface_tag, normal, root_point);
289 // Translate the root point of the normal vector along the normal direction
290 // a little bit. If the translated point is within the volume, the normal
291 // vector is inward, which has the orientation -1. If the point is outside
292 // the volume, the normal vector is outward, which has the orientation +1.
293 add_vector(root_point, normal, eps);
294
295 return is_point_in_volume(volume_tag, root_point) ? -1 : 1;
296}

299template <int dim, int spacedim>
300void
301GmshManip<dim, spacedim>::get_oriented_volume_boundaries(
302 const EntityTag volume_tag,
303 std::vector<EntityTag> &oriented_surface_tags,
304 const double eps)
305{
306 gmsh::vectorpair volume_dimtag(1);
307 volume_dimtag[0].first = dim + 1;
308 volume_dimtag[0].second = volume_tag;
309
310 gmsh::vectorpair surface_dimtag_list;
311 gmsh::model::getBoundary(volume_dimtag, surface_dimtag_list, false, false);
312
313 oriented_surface_tags.clear();
314 for (const auto &surface_dimtag : surface_dimtag_list)
315 {
316 oriented_surface_tags.push_back(
317 surface_dimtag.second * get_surface_intrinsic_orientation(
318 surface_dimtag.second, volume_tag, eps));
319 }
320}

323template <int dim, int spacedim>
324void
325GmshManip<dim, spacedim>::get_oriented_volume_boundaries(
326 const EntityTag volume_tag,
327 std::vector<EntityTag> &oriented_surface_tags,
328 std::map<EntityTag, std::array<EntityTag, 2>> &face_to_subdomain,
329 const double eps)
330{
331 gmsh::vectorpair volume_dimtag(1);
332 volume_dimtag[0].first = dim + 1;
333 volume_dimtag[0].second = volume_tag;
334
335 gmsh::vectorpair surface_dimtag_list;
336 gmsh::model::getBoundary(volume_dimtag, surface_dimtag_list, false, false);
337
338 oriented_surface_tags.clear();
339 for (const auto &surface_dimtag : surface_dimtag_list)
340 {
341 int surface_orientation;
342
343 // Here we first check if this surface has been met before.
344 typename std::map<EntityTag, std::array<EntityTag, 2>>::iterator pos =
345 face_to_subdomain.find(surface_dimtag.second);
346 if (pos != face_to_subdomain.end())
347 {
348 if (pos->second[0] != 0 && pos->second[1] == 0)
349 {
350 surface_orientation = -1;
351 pos->second[1] = volume_tag;
352 }
353 else if (pos->second[0] == 0 && pos->second[1] != 0)
354 {
355 surface_orientation = 1;
356 pos->second[0] = volume_tag;
357 }
358 else
359 {
360 // This case cannot happen.
361 surface_orientation = 0;
362 Assert(false, ExcInternalError());
363 }
364 }
365 else
366 {
367 surface_orientation =
368 get_surface_intrinsic_orientation(surface_dimtag.second,
369 volume_tag,
370 eps);
371
372 // Insert a new record in @p face_to_subdomain map.
373 if (surface_orientation == 1)
374 {
375 face_to_subdomain[surface_dimtag.second] = {{volume_tag, 0}};
376 }
377 else
378 {
379 face_to_subdomain[surface_dimtag.second] = {{0, volume_tag}};
380 }
381 }
382
383 oriented_surface_tags.push_back(surface_dimtag.second *
384 surface_orientation);
385 }
386}
