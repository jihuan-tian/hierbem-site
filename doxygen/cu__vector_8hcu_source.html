237 template <typename T>
238 HBEM_ATTR_HOST HBEM_ATTR_DEV void
239 CUDAVector<T>::fill(const size_type n_elements_for_copy,
240 const_pointer data,
241 const size_type source_offset,
242 const size_type target_offset)
243 {
244 [[maybe_unused]] const size_type allowed_n_elements_for_copy =
245 n - target_offset;
246
247 assert(n_elements_for_copy <= allowed_n_elements_for_copy);
248
249 for (size_type i = 0; i < n_elements_for_copy; i++)
250 {
251 values[target_offset + i] = data[source_offset + i];
252 }
253 }
