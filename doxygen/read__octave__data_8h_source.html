33template <typename MatrixType>
34void
35read_matrix_from_octave(std::ifstream &in,
36 const std::string &name,
37 MatrixType &matrix)
38{
39 in.seekg(0);
40
41 std::string line_buf;
42
46 while (std::getline(in, line_buf))
47 {
48 if (line_buf.compare(std::string("# name: ") + name) == 0)
49 {
55 std::getline(in, line_buf);
56 if constexpr (numbers::NumberTraits<
57 typename MatrixType::value_type>::is_complex)
58 {
59 Assert(
60 line_buf.compare("# type: complex matrix") == 0,
61 ExcMessage(
62 "Data type for the complex valued matrix to be read should be 'complex matrix'"));
63 }
64 else
65 {
66 Assert(
67 line_buf.compare("# type: matrix") == 0,
68 ExcMessage(
69 "Data type for the real valued matrix to be read should be 'matrix'"));
70 }
71
75 std::getline(in, line_buf);
76 std::smatch sm;
77 if (!std::regex_match(line_buf, sm, std::regex("# rows: (\\d+)")))
78 {
79 ExcMessage("Cannot get n_rows of the matrix!");
80 }
81 const unsigned int n_rows = std::stoi(sm.str(1));
82
86 std::getline(in, line_buf);
87 if (!std::regex_match(line_buf, sm, std::regex("# columns: (\\d+)")))
88 {
89 ExcMessage("Cannot get n_cols of the matrix!");
90 }
91 const unsigned int n_cols = std::stoi(sm.str(1));
92
93 Assert(n_rows > 0, ExcMessage("Matrix to be read has no rows!"));
94 Assert(n_cols > 0, ExcMessage("Matrix to be read has no columns!"));
95
96 matrix.reinit(n_rows, n_cols);
100 for (typename MatrixType::size_type i = 0; i < n_rows; i++)
101 {
102 std::getline(in, line_buf);
103 std::istringstream line_buf_stream(line_buf);
107 for (typename MatrixType::size_type j = 0; j < n_cols; j++)
108 {
109 line_buf_stream >> matrix(i, j);
110 }
111 }
112
116 break;
117 }
118 }
119}
