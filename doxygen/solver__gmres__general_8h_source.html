137template <typename VectorType>
138template <typename MatrixType, typename PreconditionerType>
139void
140SolverDQGMRES<VectorType>::solve(const MatrixType &A,
141 VectorType &x,
142 const VectorType &b,
143 const PreconditionerType &preconditioner)
144{
145 using real_type = typename numbers::NumberTraits<VectorNumberType>::real_type;
146
147 // At the moment, @p LAPACKFullMatrixExt or @p dealii::SparseMatrix uses
148 // @p m() and @p n() to get matrix sizes while @p HMaxtrix uses @p get_m()
149 // and @p get_n(). Because they are not consistent, we do not perform a
150 // dimension assertion about the matrix here.
151 AssertDimension(x.size(), b.size());
152 LogStream::Prefix prefix("dqgmres");
153
154 // Vector size.
155 const size_type n = b.size();
156
157 // Orthonormal basis vectors of the Krylov subspace and an additional vector
158 // which is orthogonal to the Krylov subspace. The solution vector is updated
159 // as \f$x_m = x_0 + V_m y_m\f$.
160 std::vector<VectorType> Vm(krylov_dim + 1);
161 for (size_type i = 0; i < krylov_dim + 1; i++)
162 Vm[i].reinit(n);
163
164 // The last column of the Hessenberg matrix \f$\overline{H}_m\f$. In the inner
165 // loop of this GMRES method, the Krylov subspace dimension incrementally
166 // increases from 1 to \f$m\f$, so the size of the last column of
167 // \f$\overline{H}_m\f$ should also change in principle. However, to reduce
168 // repeated memory allocation and release, we directly allocate a vector with
169 // the maximum size @p krylov_dim+1.
170 VectorType hm(krylov_dim + 1);
171
172 // Vectors which are transformed from the orthonormal basis vectors of the
173 // Krylov subspace, i.e. \f$P_m = V_m R_m^{-1}\f$. Then the solution vector is
174 // updated as \f$x_m = x_0 + P_m g_m\f$.
175 std::vector<VectorType> Pm(krylov_dim);
176 for (size_type i = 0; i < krylov_dim; i++)
177 Pm[i].reinit(n);
178
179 // Assistant vector for the error estimate when <code>ortho_hist_len <
180 // krylov_dim</code>, i.e. incomplete Gram-Schmidt orthogonalization is
181 // adopted. Therefore, we only accloate effective memory for @p Zm under this
182 // condition.
183 VectorType Zm;
184 if (ortho_hist_len < krylov_dim)
185 Zm.reinit(n);
186
187 // Givens transformation matrices used for transforming the Hessenberg matrix
188 // \f$\overline{H}_m\f$ into the upper triangular matrix \f$\overline{R}_m\f$.
189 std::vector<FullMatrix<VectorNumberType>> givens_mats(krylov_dim);
190 for (size_type i = 0; i < krylov_dim; i++)
191 givens_mats[i].reinit(2, 2);
192
193 // Initial residual vector.
194 VectorType r0(n);
195
196 // Residual norm at each step.
197 real_type r_norm = 0.;
198
199 // Temporary vector used for applying the preconditioner.
200 VectorType v_precond(n);
201
202 // The last entry of the right hand side vector \f$\overline{g}_m\f$ of size
203 // \f$m+1\f$ in the least square problem, which has been applied \f$m\f$-times
204 // of the Givens transformation.
205 VectorNumberType gamma_next;
206 // The last entry of the right hand side vector \f$g_m\f$ of size \f$m\f$ in
207 // the least square problem. \f$g_m\f$ is the first \f$m\f$ entries of
208 // \f$\overline{g}_m\f$.
209 VectorNumberType gamma_current;
210
211 // Outer loop for restarting the GMRES.
212 unsigned int restart_index = 1;
213 for (; restart_index <= control.max_steps(); restart_index++)
214 {
215 // Compute the initial residual vector.
216 if (is_left_precondition)
217 {
218 A.vmult(v_precond, x);
219 v_precond.sadd(VectorNumberType(-1.0), b);
220 preconditioner.vmult(r0, v_precond);
221 }
222 else
223 {
224 A.vmult(r0, x);
225 r0.sadd(VectorNumberType(-1.0), b);
226 }
227
228 r_norm = r0.l2_norm();
229 gamma_current = VectorNumberType(r_norm);
230 if (restart_index == 1)
231 {
232 if (control.log_result())
233 deallog << "restart step=0,residual norm=" << r_norm << std::endl;
234
235 if (r_norm < control.tolerance())
236 return;
237 }
238
239 // Clear the matrices and vectors.
240 for (size_type i = 0; i < krylov_dim + 1; i++)
241 Vm[i].reinit(n);
242
243 for (size_type i = 0; i < krylov_dim; i++)
244 {
245 Pm[i].reinit(n);
246 givens_mats[i].reinit(2, 2);
247 }
248
249 // Compute the first normalized basis vector for the Krylov subspace.
250 Vm[0] = r0;
251 Vm[0] /= gamma_current;
252
253 // When incomplete Gram-Schmidt orthogonalization is adopted, we also
254 // need to initialize the assistant vector @p Zm, which is used for an
255 // accurate error estimate.
256 if (ortho_hist_len < krylov_dim)
257 Zm = Vm[0];
258
259 // Inner loop for increasing the Krylov subspace dimension from 1 to @p m.
260 for (size_type j = 0; j < krylov_dim; j++)
261 {
262 // Perform a full or incomplete Gram-Schmidt orthogonalization.
263 if (is_left_precondition)
264 {
265 A.vmult(v_precond, Vm[j]);
266 preconditioner.vmult(Vm[j + 1], v_precond);
267 }
268 else
269 {
270 preconditioner.vmult(v_precond, Vm[j]);
271 A.vmult(Vm[j + 1], v_precond);
272 }
273
274 // Compute the last column of the Hessenberg matrix
275 // \f$\overline{H}_j\f$.
276 hm.reinit(krylov_dim + 1);
277 for (size_type i =
278 std::max(0,
279 static_cast<signed_size_type>(j) -
280 static_cast<signed_size_type>(ortho_hist_len) + 1);
281 i <= j;
282 i++)
283 {
284 hm(i) = LinAlg::inner_product_tbb(Vm[j + 1], Vm[i]);
285 Vm[j + 1].add(-hm(i), Vm[i]);
286 }
287
288 hm(j + 1) = VectorNumberType(Vm[j + 1].l2_norm());
289 Vm[j + 1] /= hm(j + 1);
290
291 // Apply historic Givens transformation to the last column of
292 // \f$\overline{H}_j\f$.
293 for (size_type i =
294 std::max(0,
295 static_cast<signed_size_type>(j) -
296 static_cast<signed_size_type>(ortho_hist_len));
297 static_cast<signed_size_type>(i) <=
298 static_cast<signed_size_type>(j) - 1;
299 i++)
300 LinAlg::apply_givens_2x2(hm, givens_mats[i], i);
301
302 // Compute the current Givens transformation matrix.
303 real_type denominator = 0.;
304 if constexpr (numbers::NumberTraits<VectorNumberType>::is_complex)
305 denominator = std::sqrt(std::abs(hm(j)) * std::abs(hm(j)) +
306 std::abs(hm(j + 1)) * std::abs(hm(j + 1)));
307 else
308 denominator = std::sqrt(hm(j) * hm(j) + hm(j + 1) * hm(j + 1));
309
310 VectorNumberType sj = hm(j + 1) / denominator;
311 VectorNumberType cj = hm(j) / denominator;
312
313 if constexpr (numbers::NumberTraits<VectorNumberType>::is_complex)
314 {
315 givens_mats[j](0, 0) = std::conj(cj);
316 givens_mats[j](0, 1) = std::conj(sj);
317 givens_mats[j](1, 0) = -sj;
318 givens_mats[j](1, 1) = cj;
319 }
320 else
321 {
322 givens_mats[j](0, 0) = cj;
323 givens_mats[j](0, 1) = sj;
324 givens_mats[j](1, 0) = -sj;
325 givens_mats[j](1, 1) = cj;
326 }
327
328 // Apply the current Givens transformation to the last column of the
329 // Hessenberg matrix and the right hand side vector.
330 hm(j) =
331 givens_mats[j](0, 0) * hm(j) + givens_mats[j](0, 1) * hm(j + 1);
332 hm(j + 1) = VectorNumberType(0.);
333
334 gamma_next = givens_mats[j](1, 0) * gamma_current;
335 gamma_current = givens_mats[j](0, 0) * gamma_current;
336
337 // Compute the vector @p pj.
338 Pm[j] = Vm[j];
339 for (size_type i =
340 std::max(0,
341 static_cast<signed_size_type>(j) -
342 static_cast<signed_size_type>(ortho_hist_len));
343 static_cast<signed_size_type>(i) <=
344 static_cast<signed_size_type>(j) - 1;
345 i++)
346 Pm[j].add(-hm(i), Pm[i]);
347
348 Pm[j] /= hm(j);
349
350 // Update the solution vector @p x.
351 if (is_left_precondition)
352 x.add(gamma_current, Pm[j]);
353 else
354 {
355 preconditioner.vmult(v_precond, Pm[j]);
356 x.add(gamma_current, v_precond);
357 }
358
359 // Compute the residual norm.
360 if (ortho_hist_len < krylov_dim)
361 {
362 // Incomplete orthogonalization is adopted.
363 Zm.sadd(givens_mats[j](1, 0), givens_mats[j](1, 1), Vm[j + 1]);
364 r_norm = std::abs(gamma_next) * Zm.l2_norm();
365 }
366 else
367 {
368 // Full orthogonalization is adopted.
369 r_norm = std::abs(gamma_next);
370 }
371
372 if (control.log_result())
373 deallog << "restart step=" << restart_index
374 << ",inner iteration=" << j << ",residual norm=" << r_norm
375 << std::endl;
376
377 // Check convergence.
378 if (r_norm <= control.tolerance())
379 return;
380 else
381 gamma_current = gamma_next;
382 }
383 }
384
385 AssertThrow(false, SolverControl::NoConvergence(restart_index, r_norm));
386}
