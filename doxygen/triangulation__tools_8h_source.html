41template <int dim, int spacedim>
42void
43build_vertex_to_cell_topology(
44 std::vector<std::vector<unsigned int>> &vertex_to_cell_topo,
45 const Triangulation<dim, spacedim> &triangulation)
46{
47 vertex_to_cell_topo.resize(triangulation.n_vertices());
48
49 // Iterate over each cell to build the topology.
50 for (const auto &cell : triangulation.active_cell_iterators())
51 {
52 for (unsigned int v = 0; v < GeometryInfo<2>::vertices_per_cell; v++)
53 {
54 vertex_to_cell_topo[cell->vertex_index(v)].push_back(
55 cell->active_cell_index());
56 }
57 }
58}

67template <typename IndexType>
68void
69print_vertex_to_cell_topology(
70 std::ostream &out,
71 const std::vector<std::vector<IndexType>> &vertex_to_cell_topo)
72{
73 unsigned int counter = 0;
74 for (const auto &vertex_to_cell : vertex_to_cell_topo)
75 {
76 out << counter << ": ";
77 for (auto cell_index : vertex_to_cell)
78 {
79 out << cell_index << " ";
80 }
81 out << std::endl;
82 counter++;
83 }

104template <int dim, int spacedim>
105void
106mark_interfacial_cells_between_materials(
107 Triangulation<dim, spacedim> &surface_triangulation,
108 const std::map<typename Triangulation<dim, spacedim>::cell_iterator,
109 typename Triangulation<dim + 1, spacedim>::face_iterator>
110 &map_from_surface_to_volume_mesh,
111 const std::vector<std::vector<unsigned int>> &vertex_to_cell_topo,
112 const std::set<types::material_id> &domain_collection1,
113 const std::set<types::material_id> &domain_collection2,
114 std::set<types::material_id> &interfacial_domain_material_ids,
115 const types::material_id material_id_shift = 10000)
116{
117 interfacial_domain_material_ids.clear();
118
119 // Get the starting cell iterator.
120 auto cell_iter = surface_triangulation.begin_active();
121
122 // Iterate over each vertex in the vertex-to-cell topology.
123 for (const auto &associated_cell_indices : vertex_to_cell_topo)
124 {
125 bool is_domain_collection1_hit = false;
126 bool is_domain_collection2_hit = false;
127
128 // Vector of flags indicating which domain collection the cells belong
129 // to.
130 std::vector<unsigned int> collection_flags(
131 associated_cell_indices.size());
132
133 // Iterate over each cell related to the current vertex and label it to
134 // which domain collection it belongs.
135 unsigned int counter = 0;
136 for (auto cell_index : associated_cell_indices)
137 {
138 cell_iter = surface_triangulation.begin_active();
139 std::advance(cell_iter, cell_index);
140 // Get the material id.
141 types::material_id mat_id = cell_iter->material_id();
142
143 // Test if the current cell belongs to the first domain collection.
144 auto found_pos = domain_collection1.find(
145 mat_id >= material_id_shift ? mat_id - material_id_shift : mat_id);
146 if (found_pos != domain_collection1.cend())
147 {
148 // The current cell belongs to the first domain collection.
149 is_domain_collection1_hit = true;
150
151 collection_flags[counter] = 1;
152 }
153 else
154 {
155 // Test if the current cell belongs to the second domain
156 // collection.
157 auto found_pos = domain_collection2.find(
158 mat_id >= material_id_shift ? mat_id - material_id_shift :
159 mat_id);
160 if (found_pos != domain_collection2.cend())
161 {
162 // The current cell belongs to the second domain collection.
163 is_domain_collection2_hit = true;
164
165 collection_flags[counter] = 2;
166 }
167 else
168 {
169 // This case should not appear, since the cell should at
170 // least belong to one of the domain collections.
171 Assert(false, ExcInternalError());
172 }
173 }
174
175 counter++;
176 }
177
178 if (is_domain_collection1_hit && is_domain_collection2_hit)
179 {
180 // The current vertex is found to be shared by cells from two domain
181 // collections. Then iterate over each cell belonging to the second
182 // domain collection and shift their material ids.
183 for (unsigned int c = 0; c < associated_cell_indices.size(); c++)
184 {
185 if (collection_flags[c] == 2)
186 {
187 cell_iter = surface_triangulation.begin_active();
188 std::advance(cell_iter, associated_cell_indices[c]);
189 types::material_id mat_id = cell_iter->material_id();
190
191 if (mat_id < material_id_shift)
192 {
193 // Set the material id of the current cell in the
194 // surface mesh.
195 cell_iter->set_material_id(mat_id + material_id_shift);
196
197 // Set the boundary id of the corresponding original
198 // face in the volume mesh.
199 auto face_iter_pos =
200 map_from_surface_to_volume_mesh.find(cell_iter);
201 Assert(
202 face_iter_pos != map_from_surface_to_volume_mesh.cend(),
203 ExcMessage(
204 "The associated face iterator in the volume triangulation cannot be found for the cell iterator in the boundary mesh!"));
205
206 auto face_iter = face_iter_pos->second;
207 face_iter->set_boundary_id(mat_id + material_id_shift);
208
209 // Insert the new id into the interfacial domain set.
210 interfacial_domain_material_ids.insert(mat_id +
211 material_id_shift);
212 }
213 }
214 }
215 }
216 }
217}
