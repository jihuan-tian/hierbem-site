137template <int spacedim, typename Number>
138void
139HBlockMatrixSkewSymmPreconditioner<spacedim, Number>::compute_lu_factorization(
140 const unsigned int fixed_rank)
141{
142 // Compute Cholesky factorization for \f$M_{11}\f$ in situ, which produces
143 // \f$L_{11}\f$.
144 if (M11->get_type() == HMatrixType::HierarchicalMatrixType)
145 {
146 M11->compute_cholesky_factorization_task_parallel(fixed_rank);
147 }
148 else
149 {
150 M11->compute_cholesky_factorization(fixed_rank);
151 }
152 M11->solve_cholesky_by_forward_substitution_matrix_valued(*M12, fixed_rank);
153 // Calculate \f$M_{22} = M_{22} + U_{12}^T U_{12}\f$
154 M12->Tmmult_level_conserving(*M22, *M12, fixed_rank, true);
155 // Apply Cholesky factorization to the new @p M22. After this operation,
156 // @p M22 stores @p L22.
157 if (M22->get_type() == HMatrixType::HierarchicalMatrixType)
158 {
159 M22->compute_cholesky_factorization_task_parallel(fixed_rank);
160 }
161 else
162 {
163 M22->compute_cholesky_factorization(fixed_rank);
164 }
165}

168template <int spacedim, typename Number>
169void
170HBlockMatrixSkewSymmPreconditioner<spacedim, Number>::vmult(
171 Vector<Number> &x,
172 const Vector<Number> &b) const
173{
174 // Verify the dimensions of matrices and vectors should match.
175 AssertDimension(x.size(), M11->get_n() + M12->get_n());
176 AssertDimension(b.size(), M11->get_m() + M22->get_m());
177
178 // Verify the dimensions of matrix blocks should match.
179 AssertDimension(M11->get_m(), M12->get_m());
180 AssertDimension(M12->get_n(), M22->get_n());
181
182 // Verify the whole block matrix should be square.
183 AssertDimension(x.size(), b.size());
184
185 // Split the vectors.
186 const size_type n1 = M11->get_n();
187 const size_type n2 = M12->get_n();
188 Vector<Number> b1(n1);
189 Vector<Number> b2(n2);
190
191 copy_vector(b1, 0, b, 0, n1);
192 copy_vector(b2, 0, b, n1, n2);
193
194 // Solve \f$Ly=b\f$.
195 Vector<Number> y1(n1);
196 Vector<Number> y2(n2);
197
198 M11->solve_cholesky_by_forward_substitution(y1, b1);
199 M12->Tvmult_add(b2, y1);
200 M22->solve_cholesky_by_forward_substitution(y2, b2);
201
202 // Solve \f$Ux=y\f$.
203 Vector<Number> x1(n1);
204 Vector<Number> x2(n2);
205 M22->solve_cholesky_by_backward_substitution(x2, y2);
206 M12->vmult_add(
207 y1, typename dealii::numbers::NumberTraits<Number>::real_type(-1.0), x2);
208 M11->solve_cholesky_by_backward_substitution(x1, y1);
209
210 // Merge the result vector.
211 copy_vector(x, 0, x1, 0, n1);
212 copy_vector(x, n1, x2, 0, n2);
213}
