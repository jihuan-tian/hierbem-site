25 void
26 infer_submatrix_block_types_from_parent_hmat(
27 const BlockType parent_hmat_block_type,
28 std::array<HMatrixSupport::BlockType, 4> &submatrix_block_types)
29 {
30 switch (parent_hmat_block_type)
31 {
32 case HMatrixSupport::diagonal_block: {
33 submatrix_block_types[0] = HMatrixSupport::diagonal_block;
34 submatrix_block_types[1] = HMatrixSupport::upper_triangular_block;
35 submatrix_block_types[2] = HMatrixSupport::lower_triangular_block;
36 submatrix_block_types[3] = HMatrixSupport::diagonal_block;
37
38 break;
39 }
40 case HMatrixSupport::lower_triangular_block: {
41 submatrix_block_types[0] = HMatrixSupport::lower_triangular_block;
42 submatrix_block_types[1] = HMatrixSupport::lower_triangular_block;
43 submatrix_block_types[2] = HMatrixSupport::lower_triangular_block;
44 submatrix_block_types[3] = HMatrixSupport::lower_triangular_block;
45
46 break;
47 }
48 case HMatrixSupport::upper_triangular_block: {
49 submatrix_block_types[0] = HMatrixSupport::upper_triangular_block;
50 submatrix_block_types[1] = HMatrixSupport::upper_triangular_block;
51 submatrix_block_types[2] = HMatrixSupport::upper_triangular_block;
52 submatrix_block_types[3] = HMatrixSupport::upper_triangular_block;
53
54 break;
55 }
56 case HMatrixSupport::undefined_block: {
57 submatrix_block_types[0] = HMatrixSupport::undefined_block;
58 submatrix_block_types[1] = HMatrixSupport::undefined_block;
59 submatrix_block_types[2] = HMatrixSupport::undefined_block;
60 submatrix_block_types[3] = HMatrixSupport::undefined_block;
61
62 break;
63 }
64 default: {
65 Assert(false,
66 ExcMessage(std::string("Invalid H-matrix types: ") +
67 std::to_string(parent_hmat_block_type)));
68
69 break;
70 }
71 }
72 }
