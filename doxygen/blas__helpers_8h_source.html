54 template <typename VectorType>
55 typename VectorType::value_type
56 inner_product_helper(const size_type n,
57 const VectorType &x,
58 const VectorType &y,
59 const size_type incx = 1,
60 const size_type incy = 1)
61 {
62 using Number = typename VectorType::value_type;
63
64 static_assert(
65 std::is_same<Number, double>::value ||
66 std::is_same<Number, float>::value ||
67 std::is_same<Number, std::complex<double>>::value ||
68 std::is_same<Number, std::complex<float>>::value,
69 "Only implemented for double, float, std::complex<double> and std::complex<float>");
70
71 // In BLAS, complex conjugation is applied to the first operand, therefore,
72 // we put y before x.
73 return inner_product(&n, y.data(), &incy, x.data(), &incx);
74 }
