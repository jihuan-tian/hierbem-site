46 template <int spacedim, typename RangeNumberType = double>
47 HBEM_ATTR_DEV void
48 collect_two_components_from_point3(
49 const HierBEM::CUDAWrappers::CUDATable<1,
50 Point<spacedim, RangeNumberType>>
51 &points,
52 const unsigned int first_component,
53 const unsigned int second_component,
54 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType>
55 &two_component_coords)
56 {
57 assert(first_component < spacedim);
58 assert(second_component < spacedim);
59 assert(two_component_coords.m() == 2);
60 assert(points.size(0) == two_component_coords.n());
61
62 for (unsigned int i = 0; i < points.size(0); i++)
63 {
64 two_component_coords(0, i) = points(i)(first_component);
65 two_component_coords(1, i) = points(i)(second_component);
66 }
67 }

140 template <int spacedim, typename RangeNumberType = double>
141 HBEM_ATTR_DEV RangeNumberType
142 surface_jacobian_det_and_normal_vector(
143 const unsigned int k3_index,
144 const unsigned int quad_no,
145 const HierBEM::CUDAWrappers::CUDATable<4, RangeNumberType>
146 &mapping_shape_grad_matrix_table,
147 const unsigned int mapping_index,
148 const unsigned int mapping_n_shape_functions,
149 const HierBEM::CUDAWrappers::CUDATable<1, Point<2, RangeNumberType>>
150 &mapping_support_points_in_real_cell_xy_components,
151 const HierBEM::CUDAWrappers::CUDATable<1, Point<2, RangeNumberType>>
152 &mapping_support_points_in_real_cell_yz_components,
153 const HierBEM::CUDAWrappers::CUDATable<1, Point<2, RangeNumberType>>
154 &mapping_support_points_in_real_cell_zx_components,
155 Tensor<1, spacedim, RangeNumberType> &normal_vector,
156 const bool is_normal_vector_negated = false)
157 {
161 assert(spacedim == 3);
162
170 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType>
171 mapping_shape_grad_matrix_at_quad_point(
172 const_cast<RangeNumberType *>(&mapping_shape_grad_matrix_table(
173 mapping_index, k3_index, quad_no, 0)),
174 mapping_n_shape_functions,
175 2);
176
177 RangeNumberType jacobian_matrix_2x2_data[4];
178 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType>
179 jacobian_matrix_2x2(jacobian_matrix_2x2_data, 2, 2);
180
181 RangeNumberType surface_jacobian_det = RangeNumberType();
182 RangeNumberType surface_jacobian_det_components[spacedim];
183
184 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType>
185 two_component_coords(
186 (RangeNumberType *)(mapping_support_points_in_real_cell_xy_components
187 .data()),
188 2,
189 mapping_n_shape_functions);
190 two_component_coords.mmult(jacobian_matrix_2x2,
191 mapping_shape_grad_matrix_at_quad_point);
192 surface_jacobian_det_components[0] = jacobian_matrix_2x2.determinant2x2();
193 surface_jacobian_det +=
194 surface_jacobian_det_components[0] * surface_jacobian_det_components[0];
195
196 two_component_coords.reinit(
197 (RangeNumberType *)(mapping_support_points_in_real_cell_yz_components
198 .data()),
199 2,
200 mapping_n_shape_functions);
201 two_component_coords.mmult(jacobian_matrix_2x2,
202 mapping_shape_grad_matrix_at_quad_point);
203 surface_jacobian_det_components[1] = jacobian_matrix_2x2.determinant2x2();
204 surface_jacobian_det +=
205 surface_jacobian_det_components[1] * surface_jacobian_det_components[1];
206
207 two_component_coords.reinit(
208 (RangeNumberType *)(mapping_support_points_in_real_cell_zx_components
209 .data()),
210 2,
211 mapping_n_shape_functions);
212 two_component_coords.mmult(jacobian_matrix_2x2,
213 mapping_shape_grad_matrix_at_quad_point);
214 surface_jacobian_det_components[2] = jacobian_matrix_2x2.determinant2x2();
215 surface_jacobian_det +=
216 surface_jacobian_det_components[2] * surface_jacobian_det_components[2];
217
218 surface_jacobian_det = std::sqrt(surface_jacobian_det);
219
225#pragma unroll
226 for (unsigned int i = 0; i < spacedim; i++)
227 {
228 if (is_normal_vector_negated)
229 {
230 normal_vector[i] =
231 -surface_jacobian_det_components[(i + 1) % spacedim] /
232 surface_jacobian_det;
233 }
234 else
235 {
236 normal_vector[i] =
237 surface_jacobian_det_components[(i + 1) % spacedim] /
238 surface_jacobian_det;
239 }
240 }
241
242 return surface_jacobian_det;
243 }

264 template <int spacedim, typename RangeNumberType = double>
265 HBEM_ATTR_DEV RangeNumberType
266 surface_jacobian_det_normal_vector_and_covariant(
267 const unsigned int k3_index,
268 const unsigned int quad_no,
269 const HierBEM::CUDAWrappers::CUDATable<4, RangeNumberType>
270 &mapping_shape_grad_matrix_table,
271 const unsigned int mapping_index,
272 const unsigned int mapping_n_shape_functions,
273 Point<spacedim, RangeNumberType> *mapping_support_points_in_real_cell,
274 Tensor<1, spacedim, RangeNumberType> &normal_vector,
275 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType> &covariant_matrix,
276 const bool is_normal_vector_negated,
277 const bool if_calculate_covariant)
278 {
282 assert(spacedim == 3);
283
291 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType>
292 mapping_shape_grad_matrix_at_quad_point(
293 const_cast<RangeNumberType *>(&mapping_shape_grad_matrix_table(
294 mapping_index, k3_index, quad_no, 0)),
295 mapping_n_shape_functions,
296 2);
297
298 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType>
299 mapping_support_point_coordinate_matrix(
300 (RangeNumberType *)(mapping_support_points_in_real_cell),
301 spacedim,
302 mapping_n_shape_functions);
303
304 RangeNumberType jacobian_matrix_data[spacedim * 2];
305 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType> jacobian_matrix(
306 jacobian_matrix_data, spacedim, 2);
307
308 mapping_support_point_coordinate_matrix.mmult(
309 jacobian_matrix, mapping_shape_grad_matrix_at_quad_point);
313 RangeNumberType G_data[4];
314 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType> G(G_data, 2, 2);
315 jacobian_matrix.Tmmult(G, jacobian_matrix);
316
317 if (if_calculate_covariant)
318 {
322 RangeNumberType G_inv_data[4];
323 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType> G_inv(
324 G_inv_data, 2, 2);
325 G_inv.invert2x2(G);
329 jacobian_matrix.mmult(covariant_matrix, G_inv);
330 }
331
332 RangeNumberType surface_jacobian_det = std::sqrt(G.determinant2x2());
333
334 if (is_normal_vector_negated)
335 {
336 normal_vector[0] =
337 (jacobian_matrix_data[2] * jacobian_matrix_data[4] -
338 jacobian_matrix_data[1] * jacobian_matrix_data[5]) /
339 surface_jacobian_det;
340 normal_vector[1] =
341 (jacobian_matrix_data[0] * jacobian_matrix_data[5] -
342 jacobian_matrix_data[2] * jacobian_matrix_data[3]) /
343 surface_jacobian_det;
344 normal_vector[2] =
345 (jacobian_matrix_data[1] * jacobian_matrix_data[3] -
346 jacobian_matrix_data[0] * jacobian_matrix_data[4]) /
347 surface_jacobian_det;
348 }
349 else
350 {
351 normal_vector[0] =
352 (jacobian_matrix_data[1] * jacobian_matrix_data[5] -
353 jacobian_matrix_data[2] * jacobian_matrix_data[4]) /
354 surface_jacobian_det;
355 normal_vector[1] =
356 (jacobian_matrix_data[2] * jacobian_matrix_data[3] -
357 jacobian_matrix_data[0] * jacobian_matrix_data[5]) /
358 surface_jacobian_det;
359 normal_vector[2] =
360 (jacobian_matrix_data[0] * jacobian_matrix_data[4] -
361 jacobian_matrix_data[1] * jacobian_matrix_data[3]) /
362 surface_jacobian_det;
363 }
364
365 return surface_jacobian_det;
366 }

394 template <int spacedim, typename RangeNumberType = double>
395 HBEM_ATTR_DEV void
396 surface_covariant_transformation(
397 const unsigned int k3_index,
398 const unsigned int quad_no,
399 const HierBEM::CUDAWrappers::CUDATable<4, RangeNumberType>
400 &mapping_shape_grad_matrix_table,
401 const unsigned int mapping_index,
402 const unsigned int mapping_n_shape_functions,
403 const HierBEM::CUDAWrappers::CUDATable<1,
404 Point<spacedim, RangeNumberType>>
405 &mapping_support_points_in_real_cell,
406 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType> &covariant)
407 {
411 assert(spacedim == 3);
412
418 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType>
419 mapping_shape_grad_matrix_at_quad_point(
420 const_cast<RangeNumberType *>(&mapping_shape_grad_matrix_table(
421 mapping_index, k3_index, quad_no, 0)),
422 mapping_n_shape_functions,
423 2);
424
425 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType>
426 all_component_coords(
427 (RangeNumberType *)(mapping_support_points_in_real_cell.data()),
428 spacedim,
429 mapping_n_shape_functions);
430
431 RangeNumberType jacobian_matrix_data[spacedim * 2];
432 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType> jacobian_matrix(
433 jacobian_matrix_data, spacedim, 2);
434 all_component_coords.mmult(jacobian_matrix,
435 mapping_shape_grad_matrix_at_quad_point);
436
437 // Gramian tensor
438 RangeNumberType G_data[4];
439 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType> G(G_data, 2, 2);
440 // Inverse of the metric tensor
441 RangeNumberType G_inv_data[4];
442 HierBEM::CUDAWrappers::CUDAFullMatrix<RangeNumberType> G_inv(G_inv_data,
443 2,
444 2);
445
446 // Compute \f$G=J^T J\f$.
447 jacobian_matrix.Tmmult(G, jacobian_matrix);
448
449 G_inv.invert2x2(G);
450
451 jacobian_matrix.mmult(covariant, G_inv);
452 }

469 template <int spacedim, typename RangeNumberType = double>
470 HBEM_ATTR_DEV void
471 transform_quad_point_from_unit_to_permuted_real_cell(
472 const unsigned int k3_index,
473 const unsigned int quad_no,
474 const HierBEM::CUDAWrappers::CUDATable<4, RangeNumberType>
475 &mapping_shape_value_table,
476 const unsigned int mapping_index,
477 const unsigned int mapping_n_shape_functions,
478 const Point<spacedim, RangeNumberType>
479 *mapping_support_points_in_real_cell,
480 Point<spacedim, RangeNumberType> &quad_point_in_real_cell)
481 {
482 // Reset all coordinate components to zero.
483 Point<spacedim, RangeNumberType> local_quad_point_in_real_cell;
484
485#pragma unroll
486 for (unsigned int i = 0; i < spacedim; i++)
487 local_quad_point_in_real_cell(i) = RangeNumberType();
488
493 for (unsigned int i = 0; i < mapping_n_shape_functions; i++)
494 {
495 local_quad_point_in_real_cell =
496 local_quad_point_in_real_cell +
497 mapping_shape_value_table(mapping_index, i, k3_index, quad_no) *
498 (*(mapping_support_points_in_real_cell + i));
499 }
500
501 quad_point_in_real_cell = local_quad_point_in_real_cell;
502 }
