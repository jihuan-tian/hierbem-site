86 void
87 get_partitions(std::vector<std::pair<int64_t, int64_t>> &parts) const
88 {
89 parts.resize(_p);
90
91 int64_t cur = 0;
92 int64_t prev = 0;
93 for (int64_t k = 0; k < _p - 1; k++)
94 {
95 while (cur < _n - (_p - (k + 1)) // ensure there are enough indexes
96 // left for the rest of partitions
97 && _f(prev, cur) <= _minmax_cost // cost of current interval is
98 // no more than the maximum cost
99 )
100 {
101 ++cur; // extend right boundary of current interval
102 }
103 parts[k] = std::make_pair(prev, cur - 1);
104 prev = cur;
105 }
106 parts[_p - 1] = std::make_pair(prev, _n - 1);
107 }

113 void
114 partition()
115 {
116 for (int64_t i = _n - 1; i >= _p - 1; i--)
117 {
118 // The first partition of optimal 1-partition for [i, n-1] is [i,
119 // n-1], so g(i, 1) = f(i, n-1)
120 _g[I(i, 1)] = _f(i, _n - 1);
121 }
122
123 for (int64_t k = 2; k <= _p; k++)
124 {
125 // The first partition for optimal k-partitions for [n-k, n-1] (the
126 // interval length is k) is [n-k, n-k], as each partition has 1
127 // element at most
128 _g[I(_n - k, k)] =
129 std::max(_f(_n - k, _n - k), _g[I(_n - k + 1, k - 1)]);
130 auto j = _n - k;
131
132 for (int64_t i = _n - k - 1; i >= _p - k; i--)
133 {
134 // The maximum cost of the optimal k-partitions for [i, n-1] will
135 // be found in [j+1, n-1], so g(i, k) = g(j+1, k-1) according to
136 // Lemma 2 in the paper
137 if (_f(i, j) <= _g[I(j + 1, k - 1)])
138 {
139 _g[I(i, k)] = _g[I(j + 1, k - 1)];
140 }
141 else
142 {
143 auto i_cost = _f(i, i);
144 if (i_cost >= _g[I(i + 1, k - 1)])
145 {
146 _g[I(i, k)] = i_cost;
147 j = i;
148 }
149 else
150 {
151 // Search for balance point in the interval [i, j]
152 while (_f(i, j - 1) >= _g[I(j, k - 1)])
153 {
154 j--;
155 }
156
157 // Current balance point is j, apply Theorem 5 in the
158 // paper
159 _g[I(i, k)] = std::min(_f(i, j), _g[I(j, k - 1)]);
160 if (_g[I(i, k)] == _g[I(j, k - 1)])
161 {
162 j--;
163 }
164 }
165 }
166 }
167 }
168
169 _minmax_cost = _g[I(0, _p)];
170 }

184 inline int64_t
185 I(int64_t i, int64_t k) const
186 {
187 return (k - 1) * _n + i;
188 }
