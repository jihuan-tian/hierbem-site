157template <int spacedim, typename Number>
158void
159HBlockMatrixSkewSymm<spacedim, Number>::vmult(Vector<Number> &y,
160 const Vector<Number> &x) const
161{
162 // Verify the dimensions of matrices and vectors should match.
163 AssertDimension(y.size(), M11->get_m() + M22->get_m());
164 AssertDimension(x.size(), M11->get_n() + M12->get_n());
165
166 // Verify the dimensions of matrix blocks should match.
167 AssertDimension(M11->get_m(), M12->get_m());
168 AssertDimension(M12->get_n(), M22->get_n());
169
170 // Verify the whole block matrix should be square.
171 AssertDimension(y.size(), x.size());
172
173 // The diagonal blocks should be symmetric.
174 Assert(M11->get_property() == HMatrixSupport::Property::symmetric,
175 ExcInvalidHMatrixProperty(M11->get_property()));
176 Assert(M22->get_property() == HMatrixSupport::Property::symmetric,
177 ExcInvalidHMatrixProperty(M22->get_property()));
178
179 // Split the input vector @p x.
180 const size_type n1 = M11->get_n();
181 const size_type n2 = M12->get_n();
182 Vector<Number> x1(n1);
183 Vector<Number> x2(n2);
184
185 copy_vector(x1, 0, x, 0, n1);
186 copy_vector(x2, 0, x, n1, n2);
187
188 // Split the result vector.
189 Vector<Number> y1(n1);
190 Vector<Number> y2(n2);
191
192 M11->vmult(y1, x1);
193 M12->vmult_add(y1, x2);
194 M12->Tvmult(y2,
195 typename dealii::numbers::NumberTraits<Number>::real_type(-1.0),
196 x1);
197 M22->vmult_add(y2, x2);
198
199 // Merge @p y1 and @p y2 into the result vector @p y.
200 copy_vector(y, 0, y1, 0, n1);
201 copy_vector(y, n1, y2, 0, n2);
202}
