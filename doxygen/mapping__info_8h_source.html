245template <int dim, int spacedim>
246void
247MappingInfo<dim, spacedim>::init_lexicographic_numberings_and_reverse()
248{
249 // Generate lexicographic numberings.
250 for (unsigned int v = 0; v < GeometryInfo<dim>::vertices_per_cell; v++)
251 {
252 lexicographic_numberings[v].resize(data->n_shape_functions);
253
254 if (v == 0)
255 {
256 lexicographic_numberings[v] =
257 FETools::lexicographic_to_hierarchic_numbering<dim>(
258 mapping.get_degree());
259 }
260 else
261 {
262 BEMTools::generate_forward_mapping_support_point_permutation(
263 mapping, v, lexicographic_numberings[v]);
264 }
265 }
266
267 // Generate reversed lexicographic numberings.
268 for (unsigned int v = 0; v < GeometryInfo<dim>::vertices_per_cell; v++)
269 {
270 reversed_lexicographic_numberings[v].resize(data->n_shape_functions);
271
272 BEMTools::generate_backward_mapping_support_point_permutation(
273 mapping, v, reversed_lexicographic_numberings[v]);
274 }
275}
