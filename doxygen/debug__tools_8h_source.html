149template <typename VectorType>
150void
151print_vector_values(std::ostream &out,
152 const VectorType &values,
153 const std::string &sep = std::string(","),
154 bool has_newline = true)
155{
156 typename VectorType::const_iterator iter_copy;
157
158 for (auto iter = values.begin(); iter != values.end(); std::advance(iter, 1))
159 {
160 iter_copy = iter;
161 std::advance(iter_copy, 1);
162
163 if (iter_copy == values.end())
164 if (has_newline)
165 {
166 out << (*iter) << std::endl;
167 }
168 else
169 {
170 out << (*iter);
171 }
172 else
173 out << (*iter) << sep;
174 }
175}

178template <typename VectorType>
179void
180print_vector_indices(std::ostream &out,
181 const VectorType &values,
182 const std::string &sep,
183 bool index_starting_from_zero,
184 bool has_newline = true)
185{
186 typename VectorType::const_iterator iter_copy;
187
188 for (auto iter = values.cbegin(); iter != values.cend();
189 std::advance(iter, 1))
190 {
191 iter_copy = iter;
192 std::advance(iter_copy, 1);
193
194 if (iter_copy == values.cend())
195 if (has_newline)
196 {
197 if (index_starting_from_zero)
198 {
199 out << (*iter) << std::endl;
200 }
201 else
202 {
203 out << (*iter) + 1 << std::endl;
204 }
205 }
206 else
207 {
208 if (index_starting_from_zero)
209 {
210 out << (*iter);
211 }
212 else
213 {
214 out << (*iter) + 1;
215 }
216 }
217 else
218 {
219 if (index_starting_from_zero)
220 {
221 out << (*iter) << sep;
222 }
223 else
224 {
225 out << (*iter) + 1 << sep;
226 }
227 }
228 }

246template <typename VectorType>
247void
248print_vector_to_mat(std::ostream &out,
249 const std::string &name,
250 const VectorType &values,
251 bool is_row_vector = false,
252 const unsigned int precision = 15,
253 const unsigned int width = 25)
254{
255 out << "# name: " << name << "\n";
256
257 if constexpr (numbers::NumberTraits<
258 typename VectorType::value_type>::is_complex)
259 out << "# type: complex matrix\n";
260 else
261 out << "# type: matrix\n";
262
263 if (is_row_vector)
264 {
265 out << "# rows: 1\n";
266 out << "# columns: " << values.size() << "\n";
267 }
268 else
269 {
270 out << "# rows: " << values.size() << "\n";
271 out << "# columns: 1\n";
272 }
273
274 for (auto iter = values.begin(); iter != values.end(); iter++)
275 {
276 out << std::setprecision(precision) << std::setw(width) << (*iter)
277 << "\n";
278 }
279
280 out << "\n\n";
281}

354template <typename T>
355void
356print_2d_table_to_mat(std::ostream &out,
357 const std::string &name,
358 const Table<2, T> &values,
359 const unsigned int precision = 8,
360 const bool scientific = true,
361 const unsigned int width = 0)
362{
363 const typename TableBase<2, T>::size_type m = values.size(0);
364 const typename TableBase<2, T>::size_type n = values.size(1);
365
366 out << "# name: " << name << "\n";
367
368 if constexpr (numbers::NumberTraits<T>::is_complex)
369 out << "# type: complex matrix\n";
370 else
371 out << "# type: matrix\n";
372
373 out << "# rows: " << m << "\n";
374 out << "# columns: " << n << "\n";
375
376 for (typename TableBase<2, T>::size_type i = 0; i < m; i++)
377 {
378 for (typename TableBase<2, T>::size_type j = 0; j < n; j++)
379 {
380 out << (scientific ? std::scientific : std::fixed) << std::setw(width)
381 << std::setprecision(precision) << values(TableIndices<2>(i, j))
382 << " ";
383 }
384 out << std::endl;
385 }
386
387 out << "\n\n";
388}

651template <int dim>
652void
653print_qgauss(std::ostream &out,
654 const QGauss<dim> &quad_cpu,
655 const std::string &separator = std::string(","))
656{
657 for (unsigned int q = 0; q < quad_cpu.size(); q++)
658 {
659 out << "#" << q << ": points=(";
660
661 for (unsigned int i = 0; i < dim; i++)
662 {
663 if (i == dim - 1)
664 {
665 out << quad_cpu.point(q)(i) << "), weights=";
666 }
667 else
668 {
669 out << quad_cpu.point(q)(i) << separator;
670 }
671 }
672
673 out << quad_cpu.weight(q) << std::endl;
674 }
