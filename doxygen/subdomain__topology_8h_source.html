95template <int dim, int spacedim>
96void
97SubdomainTopology<dim, spacedim>::generate_topology(
98 const std::string &cad_file,
99 const std::string &mesh_file,
100 const double eps_for_orientation_detection)
101{
102 gmsh::initialize();
103 gmsh::option::setNumber("General.Verbosity", 0);
104 gmsh::open(cad_file);
105 gmsh::merge(mesh_file);
106
107 // Check if the format of the CAD file is geo.
108 size_t is_geo_cad = cad_file.rfind(std::string(".geo"));
109
110 if (is_geo_cad != std::string::npos)
111 {
112 if (dim > 1)
113 throw(ExcMessage("geo model can only be used for 2D model!"));
114 else
115 gmsh::model::geo::synchronize();
116 }
117 else
118 gmsh::model::occ::synchronize();
119
120 Assert(gmsh::model::getDimension() >= dim + 1, ExcInternalError());
121
122 gmsh::vectorpair volume_dimtag_list;
123 if (is_geo_cad)
124 gmsh::model::getEntities(volume_dimtag_list, dim + 1);
125 else
126 gmsh::model::occ::getEntities(volume_dimtag_list, dim + 1);
127
128 Assert(volume_dimtag_list.size() > 0, ExcInternalError());
129
130 // The boundary entities of each volume entity.
131 std::vector<EntityTag> oriented_surface_tags;
132 for (const auto &volume_dimtag : volume_dimtag_list)
133 {
134 GmshManip<dim, spacedim>::get_oriented_volume_boundaries(
135 volume_dimtag.second,
136 oriented_surface_tags,
137 surface_to_subdomain,
138 eps_for_orientation_detection);
139
140 std::sort(oriented_surface_tags.begin(), oriented_surface_tags.end());
141 subdomain_to_surface[volume_dimtag.second] = oriented_surface_tags;
142 }
143
144 gmsh::clear();
145 gmsh::finalize();
146}

149template <int dim, int spacedim>
150void
151SubdomainTopology<dim, spacedim>::
152 generate_single_domain_topology_for_dealii_model(
153 const std::vector<EntityTag> &surface_list)
154{
155 std::array<EntityTag, 2> subdomains{{1, 0}};
156 for (auto s : surface_list)
157 {
158 subdomain_to_surface[1].push_back(s);
159 surface_to_subdomain[s] = subdomains;
160 }
161
162 std::sort(subdomain_to_surface[1].begin(), subdomain_to_surface[1].end());
163}

166template <int dim, int spacedim>
167void
168SubdomainTopology<dim, spacedim>::print(std::ostream &out) const
169{
170 out << "=== subdomain-to-face ===\n";
171
172 for (const auto &record : subdomain_to_surface)
173 {
174 out << record.first << ":";
175 for (const auto face_tag : record.second)
176 {
177 out << " " << face_tag;
178 }
179 out << "\n";
180 }
181
182 out << "=== face-to-subdomain ===\n";
183 for (const auto &record : surface_to_subdomain)
184 {
185 out << record.first << ": " << record.second[0] << " " << record.second[1]
186 << "\n";
187 }
188
189 out << std::endl;
190}

201template <int dim, int spacedim>
202class SurfaceNormalDetector
203{
204public:
205 SurfaceNormalDetector() = delete;
206
207 SurfaceNormalDetector(SubdomainTopology<dim, spacedim> &subdomain_topology)
208 : subdomain_topology(subdomain_topology)
209 {}
210
229 bool
230 is_normal_vector_inward(const types::material_id m) const
231 {
232 if (subdomain_topology.get_surface_to_subdomain()[m][0] > 0)
233 {
234 Assert(subdomain_topology.get_surface_to_subdomain()[m][1] == 0,
235 ExcInternalError());
236 return false;
237 }
238 else
239 {
240 Assert(subdomain_topology.get_surface_to_subdomain()[m][1] > 0,
241 ExcInternalError());
242 return true;
243 }
244 }
245
246private:
247 SubdomainTopology<dim, spacedim> &subdomain_topology;
248};
