33 template <int dim, typename Number1, typename Number2>
34 HBEM_ATTR_HOST
35 HBEM_ATTR_DEV Tensor<1, dim, typename ProductType<Number1, Number2>::type>
36 cross_product_3d(const Tensor<1, dim, Number1> &src1,
37 const Tensor<1, dim, Number2> &src2)
38 {
39 Tensor<1, dim, typename ProductType<Number1, Number2>::type> result;
40
41 // avoid compiler warnings
42 constexpr int s0 = 0 % dim;
43 constexpr int s1 = 1 % dim;
44 constexpr int s2 = 2 % dim;
45
46 result[s0] = src1[s1] * src2[s2] - src1[s2] * src2[s1];
47 result[s1] = src1[s2] * src2[s0] - src1[s0] * src2[s2];
48 result[s2] = src1[s0] * src2[s1] - src1[s1] * src2[s0];
49
50 return result;
51 }

63 template <int dim, typename Number1, typename Number2>
64 HBEM_ATTR_HOST HBEM_ATTR_DEV typename ProductType<Number1, Number2>::type
65 scalar_product(const Tensor<1, dim, Number1> &t1,
66 const Tensor<1, dim, Number2> &t2)
67 {
68 typename ProductType<Number1, Number2>::type result = 0.;
69
70#ifdef __CUDA_ARCH__
71# pragma unroll
72#endif
73 for (unsigned int i = 0; i < dim; i++)
74 {
75 result += t1[i] * t2[i];
76 }
77
78 return result;
79 }
