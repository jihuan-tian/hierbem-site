191template <typename Number1, typename Number2>
192constexpr bool
193is_number_comparable()
194{
195 if constexpr (numbers::NumberTraits<Number1>::is_complex &&
196 numbers::NumberTraits<Number2>::is_complex)
197 /* Both numbers are complex. */
198 return true;
199 if constexpr (!numbers::NumberTraits<Number1>::is_complex &&
200 !numbers::NumberTraits<Number2>::is_complex)
201 /* Both numbers are real. */
202 return true;
203 else if constexpr (numbers::NumberTraits<Number1>::is_complex)
204 /* Number1 is complex and Number2 is real. */
205 return type_rank<
206 typename numbers::NumberTraits<Number1>::real_type>::value >=
207 type_rank<Number2>::value;
208 else
209 /* Number2 is complex and Number1 is real. */
210 return type_rank<
211 typename numbers::NumberTraits<Number2>::real_type>::value >=
212 type_rank<Number1>::value;
213}
