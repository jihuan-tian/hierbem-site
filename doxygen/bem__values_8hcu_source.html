803template <int dim, int spacedim, typename RangeNumberType>
804void
805BEMValues<dim, spacedim, RangeNumberType>::shape_function_values_same_panel()
806{
807 const unsigned int kx_dofs_per_cell = kx_fe.dofs_per_cell;
808 const unsigned int ky_dofs_per_cell = ky_fe.dofs_per_cell;
809 [[maybe_unused]] const unsigned int max_mapping_n_shape_functions =
810 mappings.back()->get_data()->n_shape_functions;
811 const unsigned int n_q_points = quad_rule_for_same_panel.size();
812
813 // Make assertion about the length for each dimension of the data table.
814 AssertDimension(kx_shape_value_table_for_same_panel.size(0),
815 kx_dofs_per_cell);
816 AssertDimension(kx_shape_value_table_for_same_panel.size(1), 8);
817 AssertDimension(kx_shape_value_table_for_same_panel.size(2), n_q_points);
818
819 AssertDimension(ky_shape_value_table_for_same_panel.size(0),
820 ky_dofs_per_cell);
821 AssertDimension(ky_shape_value_table_for_same_panel.size(1), 8);
822 AssertDimension(ky_shape_value_table_for_same_panel.size(2), n_q_points);
823
824 AssertDimension(kx_mapping_shape_value_table_for_same_panel.size(0),
825 mappings.size());
826 AssertDimension(kx_mapping_shape_value_table_for_same_panel.size(1),
827 max_mapping_n_shape_functions);
828 AssertDimension(kx_mapping_shape_value_table_for_same_panel.size(2), 8);
829 AssertDimension(kx_mapping_shape_value_table_for_same_panel.size(3),
830 n_q_points);
831
832 AssertDimension(ky_mapping_shape_value_table_for_same_panel.size(0),
833 mappings.size());
834 AssertDimension(ky_mapping_shape_value_table_for_same_panel.size(1),
835 max_mapping_n_shape_functions);
836 AssertDimension(ky_mapping_shape_value_table_for_same_panel.size(2), 8);
837 AssertDimension(ky_mapping_shape_value_table_for_same_panel.size(3),
838 n_q_points);
839
840 AssertDimension(kx_shape_grad_matrix_table_for_same_panel.size(0), 8);
841 AssertDimension(kx_shape_grad_matrix_table_for_same_panel.size(1),
842 n_q_points);
843
844 AssertDimension(ky_shape_grad_matrix_table_for_same_panel.size(0), 8);
845 AssertDimension(ky_shape_grad_matrix_table_for_same_panel.size(1),
846 n_q_points);
847
848 AssertDimension(kx_mapping_shape_grad_matrix_table_for_same_panel.size(0),
849 mappings.size());
850 AssertDimension(kx_mapping_shape_grad_matrix_table_for_same_panel.size(1), 8);
851 AssertDimension(kx_mapping_shape_grad_matrix_table_for_same_panel.size(2),
852 n_q_points);
853
854 AssertDimension(ky_mapping_shape_grad_matrix_table_for_same_panel.size(0),
855 mappings.size());
856 AssertDimension(ky_mapping_shape_grad_matrix_table_for_same_panel.size(1), 8);
857 AssertDimension(ky_mapping_shape_grad_matrix_table_for_same_panel.size(2),
858 n_q_points);
859
863 resize_internal_data_in_mappings(n_q_points);
864
868 std::vector<Point<dim * 2>> quad_points =
869 quad_rule_for_same_panel.get_points();
870
878 const FE_Poly_short &kx_fe_poly = dynamic_cast<const FE_Poly_short &>(kx_fe);
879 const FE_Poly_short &ky_fe_poly = dynamic_cast<const FE_Poly_short &>(ky_fe);
880
881 std::vector<unsigned int> kx_poly_space_inverse_numbering(
882 kx_fe_poly.get_poly_space_numbering_inverse());
883 std::vector<unsigned int> ky_poly_space_inverse_numbering(
884 ky_fe_poly.get_poly_space_numbering_inverse());
885
890 std::vector<Point<dim>> kx_unit_quad_points(n_q_points);
891 std::vector<Point<dim>> ky_unit_quad_points(n_q_points);
892
893 // Iterate over each $k_3$ part.
894 for (unsigned k = 0; k < 8; k++)
895 {
896 // Iterate over each quadrature point.
897 for (unsigned int q = 0; q < n_q_points; q++)
898 {
899 // Transform the quadrature point in the parametric space to
900 // the unit cells of \f$K_x\f$ and \f$K_y\f$.
901 sauter_same_panel_parametric_coords_to_unit_cells(
902 quad_points[q], k, kx_unit_quad_points[q], ky_unit_quad_points[q]);
903
909 for (unsigned int s = 0; s < kx_dofs_per_cell; s++)
910 {
911 kx_shape_value_table_for_same_panel(TableIndices<3>(s, k, q)) =
912 static_cast<RangeNumberType>(
913 kx_fe.shape_value(kx_poly_space_inverse_numbering[s],
914 kx_unit_quad_points[q]));
915 }
916
922 for (unsigned int s = 0; s < ky_dofs_per_cell; s++)
923 {
924 ky_shape_value_table_for_same_panel(TableIndices<3>(s, k, q)) =
925 static_cast<RangeNumberType>(
926 ky_fe.shape_value(ky_poly_space_inverse_numbering[s],
927 ky_unit_quad_points[q]));
928 }
929
930 // Calculate the Jacobian matrix evaluated at
931 // <code>kx_quad_point</code> in \f$K_x\f$. Matrix rows correspond
932 // to shape functions which are in the lexicographic order.
933 kx_shape_grad_matrix_table_for_same_panel(TableIndices<2>(k, q)) =
934 BEMTools::shape_grad_matrix_in_lexicographic_order<dim,
935 spacedim,
936 RangeNumberType>(
937 kx_fe, kx_unit_quad_points[q]);
938 // Calculate the Jacobian matrix evaluated at
939 // <code>ky_quad_point</code> in \f$K_y\f$. Matrix rows correspond
940 // to shape functions which are in the lexicographic order.
941 ky_shape_grad_matrix_table_for_same_panel(TableIndices<2>(k, q)) =
942 BEMTools::shape_grad_matrix_in_lexicographic_order<dim,
943 spacedim,
944 RangeNumberType>(
945 ky_fe, ky_unit_quad_points[q]);
946 }
947
962 for (unsigned int m = 0; m < mappings.size(); m++)
963 {
964 auto &mapping_data = mappings[m]->get_data();
965 const unsigned int mapping_n_shape_functions =
966 mapping_data->n_shape_functions;
967 // The numbering used for accessing shape function values and their
968 // derivatives in the lexicographic order.
969 const std::vector<unsigned int> &lexicographic_numbering =
970 mappings[m]->get_lexicographic_numberings()[0];
971
972 // Compute shape function values and their derivatives for kx.
973 mapping_data->compute_shape_function_values(kx_unit_quad_points);
974 for (unsigned int q = 0; q < n_q_points; q++)
975 {
976 for (unsigned int s = 0; s < mapping_n_shape_functions; s++)
977 {
978 kx_mapping_shape_value_table_for_same_panel(
979 TableIndices<4>(m, s, k, q)) =
980 static_cast<RangeNumberType>(
981 mapping_data->shape(q, lexicographic_numbering[s]));
982
983 for (unsigned int d = 0; d < dim; d++)
984 {
985 kx_mapping_shape_grad_matrix_table_for_same_panel(
986 TableIndices<3>(m, k, q))(s, d) =
987 static_cast<RangeNumberType>(mapping_data->derivative(
988 q, lexicographic_numbering[s])[d]);
989 }
990 }
991 }
992
993 // Compute shape function values and their derivatives for ky.
994 mapping_data->compute_shape_function_values(ky_unit_quad_points);
995 for (unsigned int q = 0; q < n_q_points; q++)
996 {
997 for (unsigned int s = 0; s < mapping_n_shape_functions; s++)
