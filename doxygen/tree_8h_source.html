756template <typename T, std::size_t N>
757TreeNode<T, N>::TreeNode(const T &data,
758 unsigned int level,
759 const std::array<TreeNode *, N> &children,
760 TreeNode *parent,
761 TreeNodeSplitMode split_mode)
762 : data(data)
763 , level(level)
764 , children(children)
765 , parent(parent)
766 , child_num(0)
767 , split_mode(split_mode)
768{
772 for (auto child : this->children)
773 {
774 if (child != nullptr)
775 {
776 child_num++;
777 }
778 }
779
780 if (child_num == 4)
781 {
787 split_mode = CrossSplitMode;
788 }
789
793 if (parent != nullptr)
794 {
795 (parent->child_num)++;
796 }
797}

1106template <typename T>
1107unsigned int
1108calc_depth(const BinaryTreeNode<T> *p)
1109{
1110 if (p == nullptr)
1111 {
1112 return 0;
1113 }
1114 else
1115 {
1116 return std::max(calc_depth(p->Left()), calc_depth(p->Right())) + 1;
1117 }

1123template <typename T, std::size_t N>
1124unsigned int
1125calc_depth(TreeNode<T, N> *p)
1126{
1127 if (p == nullptr)
1128 {
1129 return 0;
1130 }
1131 else
1132 {
1133 // Array storing the depth of all subtrees.
1134 std::array<unsigned int, N> subtree_depths;
1135
1136 for (std::size_t i = 0; i < N; i++)
1137 {
1138 subtree_depths.at(i) = calc_depth(p->get_child_pointer(i));
1139 }
1140
1141 return (*std::max_element(subtree_depths.cbegin(),
1142 subtree_depths.cend())) +
1143 1;
1144 }
1145}
